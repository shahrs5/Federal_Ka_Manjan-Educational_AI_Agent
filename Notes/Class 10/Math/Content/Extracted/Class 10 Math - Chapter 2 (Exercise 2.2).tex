\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Biquadratic:} if $ax^4+bx^2+c=0$, set $t=x^2 \Rightarrow at^2+bt+c=0$.
\item \textbf{Power/exponential:} if $ak^{2x}+bk^x+c=0$, set $t=k^x \Rightarrow at^2+bt+c=0$.
\item \textbf{Reciprocal identities (for $x\neq 0$):}
\[
\left(x+\frac1x\right)^2=x^2+\frac1{x^2}+2,\qquad
\left(x-\frac1x\right)^2=x^2+\frac1{x^2}-2.
\]
\item \textbf{Common trick:} pair factors to get the same quadratic expression, e.g.
\[
(x+a)(x+b)=x^2+(a+b)x+ab.
\]
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1 (i)}
\textcolor{gold}{\bfseries Question:} $ax^4+bx^2+c=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x^2.\\
\Step{2}\;& ax^4+bx^2+c=0 \;\Rightarrow\; a(t^2)+b(t)+c=0.\\
\Step{3}\;& \Rightarrow\; \boxed{at^2+bt+c=0 \text{ (quadratic in }t\text{)}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ii)}
\textcolor{gold}{\bfseries Question:} $9x^6-3x^3+7=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x^3.\\
\Step{2}\;& 9x^6-3x^3+7=0 \;\Rightarrow\; 9t^2-3t+7=0.\\
\Step{3}\;& \Rightarrow\; \boxed{9t^2-3t+7=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iii)}
\textcolor{gold}{\bfseries Question:} $3x+\dfrac{4}{6x-2}=-1$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq \tfrac13$)
\[
\begin{aligned}
\Step{1}\;& 3x+\frac{4}{6x-2}=-1 \;\Rightarrow\; 3x(6x-2)+4=-1(6x-2).\\
\Step{2}\;& 18x^2-6x+4=-6x+2.\\
\Step{3}\;& 18x^2+2=0 \;\Rightarrow\; \boxed{9x^2+1=0 \text{ (quadratic in }x\text{)}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iv)}
\textcolor{gold}{\bfseries Question:} $(x+1)^2+\dfrac{3}{(x+1)^2}=4$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq -1$)
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=(x+1)^2 \; (t>0).\\
\Step{2}\;& t+\frac{3}{t}=4 \;\Rightarrow\; t^2+3=4t.\\
\Step{3}\;& \Rightarrow\; \boxed{t^2-4t+3=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (v)}
\textcolor{gold}{\bfseries Question:} $3\!\left(x^2+\dfrac1{x^2}\right)+8\!\left(x+\dfrac1x\right)+11=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x+\frac1x.\\
\Step{2}\;& x^2+\frac1{x^2}=t^2-2.\\
\Step{3}\;& 3(t^2-2)+8t+11=0 \;\Rightarrow\; \boxed{3t^2+8t+5=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (vi)}
\textcolor{gold}{\bfseries Question:} $3x^4+7x^3+5x^2-7x+3=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& \frac{3x^4+7x^3+5x^2-7x+3}{x^2}=0
\;\Rightarrow\;
3\!\left(x^2+\frac1{x^2}\right)+7\!\left(x-\frac1x\right)+5=0.\\
\Step{2}\;& \text{Let } t=x-\frac1x,\quad x^2+\frac1{x^2}=t^2+2.\\
\Step{3}\;& 3(t^2+2)+7t+5=0 \;\Rightarrow\; \boxed{3t^2+7t+11=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (vii)}
\textcolor{gold}{\bfseries Question:} $ak^{2x}+bk^x+c=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=k^x.\\
\Step{2}\;& ak^{2x}+bk^x+c=0 \;\Rightarrow\; at^2+bt+c=0.\\
\Step{3}\;& \Rightarrow\; \boxed{at^2+bt+c=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (viii)}
\textcolor{gold}{\bfseries Question:} $8\cdot 4^x-7\cdot 2^x-1=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 4^x=(2^2)^x=2^{2x}=(2^x)^2.\\
\Step{2}\;& \text{Let } t=2^x \;(t>0).\\
\Step{3}\;& 8t^2-7t-1=0 \;\Rightarrow\; \boxed{8t^2-7t-1=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ix)}
\textcolor{gold}{\bfseries Question:} $(x-1)(x-2)(x+3)(x-6)=6$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (x-1)(x-6)=x^2-7x+6,\quad (x-2)(x+3)=x^2+x-6.\\
\Step{2}\;& x^2-7x+6=(x^2-3x)-(4x-6),\\
& x^2+x-6=(x^2-3x)+(4x-6).\\
\Step{3}\;& \text{Let } y=x^2-3x. \;\Rightarrow\;
(x^2-7x+6)(x^2+x-6)=\bigl(y-(4x-6)\bigr)\bigl(y+(4x-6)\bigr)\\
&=y^2-(4x-6)^2.\\
\Step{4}\;& (4x-6)^2=16x^2-48x+36=16(x^2-3x)+36=16y+36.\\
\Step{5}\;& y^2-(16y+36)=6 \;\Rightarrow\; \boxed{y^2-16y-42=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (x)}
\textcolor{gold}{\bfseries Question:} $(2x-1)(2x-7)(x-3)(x-1)=8$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (2x-1)(2x-7)=4x^2-16x+7,\quad (x-3)(x-1)=x^2-4x+3.\\
\Step{2}\;& \text{Let } y=x^2-4x.\\
\Step{3}\;& 4x^2-16x+7=4y+7,\quad x^2-4x+3=y+3.\\
\Step{4}\;& (4y+7)(y+3)=8 \;\Rightarrow\; \boxed{4y^2+19y+13=0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (i)}
\textcolor{gold}{\bfseries Question:} $x^4-20x^2+64=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x^2. \Rightarrow t^2-20t+64=0.\\
\Step{2}\;& (t-16)(t-4)=0 \Rightarrow t=16 \text{ or } t=4.\\
\Step{3}\;& x^2=16 \Rightarrow x=\pm 4,\qquad x^2=4 \Rightarrow x=\pm 2.\\
&\boxed{x=\pm 4,\ \pm 2.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (ii)}
\textcolor{gold}{\bfseries Question:} $x^4+16x^2-225=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x^2. \Rightarrow t^2+16t-225=0.\\
\Step{2}\;& t=\frac{-16\pm\sqrt{16^2+4\cdot 225}}{2}
=\frac{-16\pm\sqrt{1156}}{2}
=\frac{-16\pm 34}{2}.\\
\Step{3}\;& t=9 \text{ or } t=-25.\\
\Step{4}\;& x^2=9 \Rightarrow x=\pm 3,\quad x^2=-25 \Rightarrow x=\pm 5i.\\
&\boxed{x=\pm 3,\ \pm 5i.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (iii)}
\textcolor{gold}{\bfseries Question:} $x^{\frac{2}{5}}+5x^{\frac{1}{5}}+6=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x^{\frac15}. \Rightarrow t^2+5t+6=0.\\
\Step{2}\;& (t+2)(t+3)=0 \Rightarrow t=-2 \text{ or } t=-3.\\
\Step{3}\;& x=t^5 \Rightarrow x=(-2)^5=-32,\quad x=(-3)^5=-243.\\
&\boxed{x=-32,\ -243.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (iv)}
\textcolor{gold}{\bfseries Question:} $3x^2+\dfrac{4}{x^2}=7$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& 3x^2+\frac{4}{x^2}=7 \Rightarrow 3x^4-7x^2+4=0.\\
\Step{2}\;& \text{Let } t=x^2. \Rightarrow 3t^2-7t+4=0.\\
\Step{3}\;& (3t-4)(t-1)=0 \Rightarrow t=\frac{4}{3}\ \text{or}\ t=1.\\
\Step{4}\;& x^2=\frac43 \Rightarrow x=\pm \frac{2}{\sqrt3}=\pm \frac{2\sqrt3}{3},\quad x^2=1 \Rightarrow x=\pm 1.\\
&\boxed{x=\pm 1,\ \pm \frac{2\sqrt3}{3}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (v)}
\textcolor{gold}{\bfseries Question:} $5(x+1)+\dfrac{3}{x+1}=8$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq -1$)
\[
\begin{aligned}
\Step{1}\;& \text{Let } t=x+1\ (t\neq 0). \Rightarrow 5t+\frac{3}{t}=8.\\
\Step{2}\;& 5t^2-8t+3=0.\\
\Step{3}\;& (5t-3)(t-1)=0 \Rightarrow t=\frac35 \text{ or } t=1.\\
\Step{4}\;& x=t-1 \Rightarrow x=-\frac{2}{5}\ \text{or}\ x=0.\\
&\boxed{x=-\frac25,\ 0.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (vi)}
\textcolor{gold}{\bfseries Question:} $5x^2+\dfrac{36}{5x^2+4}=16$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } y=5x^2+4 \Rightarrow 5x^2=y-4.\\
\Step{2}\;& (y-4)+\frac{36}{y}=16 \Rightarrow y^2-4y+36=16y.\\
\Step{3}\;& y^2-20y+36=0 \Rightarrow y=\frac{20\pm 16}{2}=18 \text{ or } 2.\\
\Step{4}\;& 5x^2+4=18 \Rightarrow x^2=\frac{14}{5}\Rightarrow x=\pm \sqrt{\frac{14}{5}}=\pm\frac{\sqrt{70}}{5}.\\
&\text{Also } 5x^2+4=2 \Rightarrow x^2=-\frac{2}{5}\Rightarrow x=\pm i\sqrt{\frac{2}{5}}.\\
&\boxed{x=\pm \frac{\sqrt{70}}{5},\ \pm i\sqrt{\frac{2}{5}}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (vii)}
\textcolor{gold}{\bfseries Question:} $2x^4-x^3-6x^2-x+2=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& \frac{2x^4-x^3-6x^2-x+2}{x^2}=0
\Rightarrow 2\!\left(x^2+\frac1{x^2}\right)-\left(x+\frac1x\right)-6=0.\\
\Step{2}\;& \text{Let } t=x+\frac1x,\quad x^2+\frac1{x^2}=t^2-2.\\
\Step{3}\;& 2(t^2-2)-t-6=0 \Rightarrow 2t^2-t-10=0.\\
\Step{4}\;& t=\frac{1\pm 9}{4} \Rightarrow t=\frac52 \text{ or } t=-2.\\
\Step{5}\;& x+\frac1x=\frac52 \Rightarrow 2x^2-5x+2=0 \Rightarrow x=2,\ \frac12.\\
\Step{6}\;& x+\frac1x=-2 \Rightarrow x^2+2x+1=0 \Rightarrow x=-1.\\
&\boxed{x=2,\ \frac12,\ -1.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (viii)}
\textcolor{gold}{\bfseries Question:} $12x^4+11x^3-146x^2+11x+12=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& \frac{12x^4+11x^3-146x^2+11x+12}{x^2}=0\\
&\Rightarrow 12\!\left(x^2+\frac1{x^2}\right)+11\!\left(x+\frac1x\right)-146=0.\\
\Step{2}\;& \text{Let } t=x+\frac1x,\quad x^2+\frac1{x^2}=t^2-2.\\
\Step{3}\;& 12(t^2-2)+11t-146=0 \Rightarrow 12t^2+11t-170=0.\\
\Step{4}\;& t=\frac{-11\pm \sqrt{11^2+4\cdot 12\cdot 170}}{24}
=\frac{-11\pm \sqrt{8281}}{24}
=\frac{-11\pm 91}{24}.\\
\Step{5}\;& t=\frac{10}{3}\ \text{or}\ t=-\frac{17}{4}.\\
\Step{6}\;& x+\frac1x=\frac{10}{3}\Rightarrow 3x^2-10x+3=0 \Rightarrow x=3,\ \frac13.\\
\Step{7}\;& x+\frac1x=-\frac{17}{4}\Rightarrow 4x^2+17x+4=0 \Rightarrow x=-4,\ -\frac14.\\
&\boxed{x=3,\ \frac13,\ -4,\ -\frac14.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (ix)}
\textcolor{gold}{\bfseries Question:} $4\!\left(x^2+\dfrac1{x^2}\right)-\left(x-\dfrac1x\right)-11=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& \text{Let } u=x-\frac1x. \\
\Step{2}\;& u^2=x^2+\frac1{x^2}-2 \Rightarrow x^2+\frac1{x^2}=u^2+2.\\
\Step{3}\;& 4(u^2+2)-u-11=0 \Rightarrow 4u^2-u-3=0.\\
\Step{4}\;& u=\frac{1\pm 7}{8}\Rightarrow u=1\ \text{or}\ u=-\frac34.\\
\Step{5}\;& x-\frac1x=1 \Rightarrow x^2-x-1=0 \Rightarrow x=\frac{1\pm \sqrt5}{2}.\\
\Step{6}\;& x-\frac1x=-\frac34 \Rightarrow 4x^2+3x-4=0 \Rightarrow x=\frac{-3\pm \sqrt{73}}{8}.\\
&\boxed{x=\frac{1\pm \sqrt5}{2},\ \frac{-3\pm \sqrt{73}}{8}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (x)}

\begin{QAPair}{Question 2 (xiii)}
\textcolor{gold}{\bfseries Question:} $(x+2)(x-3)(x+10)(x+5)=-396$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (x+2)(x+5)=x^2+7x+10,\quad (x-3)(x+10)=x^2+7x-30.\\
\Step{2}\;& \text{Let } y=x^2+7x. \Rightarrow (y+10)(y-30)=-396.\\
\Step{3}\;& y^2-20y-300=-396 \Rightarrow y^2-20y+96=0.\\
\Step{4}\;& y=\frac{20\pm \sqrt{400-384}}{2}=\frac{20\pm 4}{2}\Rightarrow y=12 \text{ or } 8.\\
\Step{5}\;& x^2+7x=12 \Rightarrow x^2+7x-12=0 \Rightarrow x=\frac{-7\pm \sqrt{97}}{2}.\\
\Step{6}\;& x^2+7x=8 \Rightarrow x^2+7x-8=0 \Rightarrow x=1,\ -8.\\
&\boxed{x=1,\ -8,\ \frac{-7\pm \sqrt{97}}{2}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (xiv)}
\textcolor{gold}{\bfseries Question:} $x(x-1)(x+2)(x+3)=40$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& x(x+3)=x^2+3x,\quad (x-1)(x+2)=x^2+x-2.\\
\Step{2}\;& (x^2+3x)(x^2+x-2)=40.\\
\Step{3}\;& \text{Let } u=x^2+2x. \\
\Step{4}\;& x^2+3x=u+x,\quad x^2+x-2=u-x-2.\\
\Step{5}\;& (u+x)(u-x-2)=u^2-2u-(x^2+2x)=u^2-2u-u=u^2-3u.\\
\Step{6}\;& u^2-3u=40 \Rightarrow u^2-3u-40=0 \Rightarrow (u-8)(u+5)=0.\\
\Step{7}\;& u=8 \Rightarrow x^2+2x-8=0 \Rightarrow x=2,\ -4.\\
&\text{(Also }u=-5 \Rightarrow x^2+2x+5=0 \Rightarrow x=-1\pm 2i.)\\
&\boxed{x=2,\ -4,\ -1\pm 2i.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (xv)}
\textcolor{gold}{\bfseries Question:} $(x-2)(x-6)(x+4)(x+8)+256=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (x-2)(x+4)=x^2+2x-8,\quad (x-6)(x+8)=x^2+2x-48.\\
\Step{2}\;& \text{Let } u=x^2+2x.\\
\Step{3}\;& (u-8)(u-48)+256=0 \Rightarrow u^2-56u+384+256=0.\\
\Step{4}\;& u^2-56u+640=0 \Rightarrow u=\frac{56\pm \sqrt{56^2-4\cdot 640}}{2}
=\frac{56\pm \sqrt{576}}{2}
=\frac{56\pm 24}{2}.\\
\Step{5}\;& u=40 \text{ or } u=16.\\
\Step{6}\;& x^2+2x-40=0 \Rightarrow x=-1\pm \sqrt{41}.\\
\Step{7}\;& x^2+2x-16=0 \Rightarrow x=-1\pm \sqrt{17}.\\
&\boxed{x=-1\pm \sqrt{41},\ -1\pm \sqrt{17}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (i)}
\textcolor{gold}{\bfseries Question:} $2x^4-3x^2+1=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 2x^4-3x^2+1=(2x^2-1)(x^2-1).\\
\Step{2}\;& (2x^2-1)(x-1)(x+1)=0.\\
\Step{3}\;& 2x^2-1=0 \Rightarrow x=\pm \frac{1}{\sqrt2},\quad x^2-1=0 \Rightarrow x=\pm 1.\\
&\boxed{x=\pm 1,\ \pm \frac{1}{\sqrt2}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (ii)}
\textcolor{gold}{\bfseries Question:} $8x^6-7x^3-1=0$ (Find only real roots)\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 8x^6-7x^3-1=(8x^3+1)(x^3-1).\\
\Step{2}\;& (8x^3+1)(x-1)(x^2+x+1)=0.\\
\Step{3}\;& \text{Real roots come from }x^3=1 \Rightarrow x=1,\ \text{and } 8x^3=-1 \Rightarrow x=-\frac12.\\
&\boxed{x=1,\ -\frac12.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iii)}
\textcolor{gold}{\bfseries Question:} $x^2+\dfrac{1}{x^2}=2$\\
\tcblower
\textcolor{green}{\bfseries Answer:} (with $x\neq 0$)
\[
\begin{aligned}
\Step{1}\;& x^2+\frac{1}{x^2}=2 \Rightarrow x^4-2x^2+1=0.\\
\Step{2}\;& x^4-2x^2+1=(x^2-1)^2=0.\\
\Step{3}\;& x^2=1 \Rightarrow x=\pm 1.\\
&\boxed{x=\pm 1.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iv)}
\textcolor{gold}{\bfseries Question:} $4\cdot 2^{2x}-4\cdot 2^x+1=0$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 4\cdot 2^{2x}-4\cdot 2^x+1=\bigl(2\cdot 2^x-1\bigr)^2.\\
\Step{2}\;& \bigl(2\cdot 2^x-1\bigr)^2=0 \Rightarrow 2\cdot 2^x-1=0.\\
\Step{3}\;& 2^{x+1}=1 \Rightarrow x+1=0 \Rightarrow \boxed{x=-1.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} Write two examples of equations that are not quadratic, but can be written in quadratic form.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\begin{itemize}
\item Example 1: $x^4-5x^2+6=0$ \; (let $t=x^2 \Rightarrow t^2-5t+6=0$).
\item Example 2: $2^{2x}-7\cdot 2^x+10=0$ \; (let $t=2^x \Rightarrow t^2-7t+10=0$).
\end{itemize}
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} How would you solve $(y-4)^2-7(y-4)=-6$? Write explanation and then solve.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Notice the repeated expression }(y-4). \text{ Let } t=y-4.\\
\Step{2}\;& t^2-7t=-6 \Rightarrow t^2-7t+6=0.\\
\Step{3}\;& (t-1)(t-6)=0 \Rightarrow t=1 \text{ or } t=6.\\
\Step{4}\;& y-4=1 \Rightarrow y=5,\qquad y-4=6 \Rightarrow y=10.\\
&\boxed{y=5,\ 10.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} Solve $y^3=125$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 125=5^3.\\
\Step{2}\;& y^3=5^3 \Rightarrow \boxed{y=5} \quad (\text{real solution}).
\end{aligned}
\]
\end{QAPair}
