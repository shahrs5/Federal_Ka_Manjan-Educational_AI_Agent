\begin{QuickBox}
{\color{cyan}\bfseries Quick facts (very useful)}\par\medskip
\begin{itemize}
\item \textbf{Square root of negative:} $\sqrt{-a}=i\sqrt{a}$ \; (for $a>0$).
\item \textbf{Add/Subtract:} combine real parts together, imaginary parts together.
\item \textbf{Multiply:} use FOIL; remember $i^2=-1$.
\item \textbf{Powers of $i$:} $i^1=i$, $i^2=-1$, $i^3=-i$, $i^4=1$ (cycle repeats every 4).
\item \textbf{Rationalize denominator:} multiply by the \textbf{conjugate} $(a+bi)\mapsto(a-bi)$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1 (i)}
\textcolor{gold}{\bfseries Question:} $\sqrt{-3}$ \hfill (Express in terms of $i$)
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-3}=\sqrt{-1\cdot 3}\\
\Step{2}\;&=\sqrt{-1}\,\sqrt{3}\\
\Step{3}\;&= i\sqrt{3}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ii)}
\textcolor{gold}{\bfseries Question:} $6\sqrt{-4}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-4}=i\sqrt{4}\\
\Step{2}\;&=2i\\
\Step{3}\;& 6\sqrt{-4}=6(2i)=12i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iii)}
\textcolor{gold}{\bfseries Question:} $\sqrt{-\dfrac{4}{9}}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-\frac{4}{9}}= i\sqrt{\frac{4}{9}}\\
\Step{2}\;&= i\cdot \frac{2}{3}=\frac{2}{3}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iv)}
\textcolor{gold}{\bfseries Question:} $-\sqrt{-20}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-20}=i\sqrt{20}\\
\Step{2}\;&=i\sqrt{4\cdot 5}=i(2\sqrt5)=2i\sqrt5\\
\Step{3}\;& -\sqrt{-20}=-(2i\sqrt5)=-2i\sqrt5.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (v)}
\textcolor{gold}{\bfseries Question:} $4-\sqrt{-60}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-60}=i\sqrt{60}\\
\Step{2}\;&=i\sqrt{4\cdot 15}=i(2\sqrt{15})=2i\sqrt{15}\\
\Step{3}\;& 4-\sqrt{-60}=4-2i\sqrt{15}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (vi)}
\textcolor{gold}{\bfseries Question:} $\sqrt{-8}\,\sqrt{-2}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt{-8}=i\sqrt{8}=i(2\sqrt2)=2i\sqrt2\\
\Step{2}\;& \sqrt{-2}=i\sqrt2\\
\Step{3}\;& \sqrt{-8}\sqrt{-2}=(2i\sqrt2)(i\sqrt2)=2\,i^2\,(\sqrt2\cdot\sqrt2)\\
\Step{4}\;&=2(-1)(2)=-4.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (v)}
\textcolor{gold}{\bfseries Question:} $(2+4i)(1+2i)$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}

\Step{2}\;&=2+4i+4i+8i^2\\
\Step{3}\;&=2+8i+8(-1)=2+8i-8\\
\Step{4}\;&=-6+8i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (vi)}

\Step{2}\;&=2-3i-8i+12i^2\\
\Step{3}\;&=2-11i+12(-1)=2-11i-12\\
\Step{4}\;&=-10-11i.

\]
\end{QAPair}

\begin{QAPair}{Question 2 (vii)}

\Step{1}\;& -8i(2-2i)=(-8i)\cdot2+(-8i)\cdot(-2i)\\

\Step{3}\;&=-16i+16(-1)=-16-16i.

\]
\end{QAPair}

\begin{QAPair}{Question 2 (viii)}

\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}

\begin{QAPair}{Question 2 (ix)}

\Step{2}\;&=25+15i+15i+9i^2\\
\Step{3}\;&=25+30i+9(-1)=25+30i-9\\
\Step{4}\;&=16+30i.

\begin{QAPair}{Question 2 (xi)}
\textcolor{gold}{\bfseries Question:} $(1+\sqrt2\,i)(1-\sqrt3\,i)$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (1+\sqrt2\,i)(1-\sqrt3\,i)=1\cdot1+1\cdot(-\sqrt3\,i)+(\sqrt2\,i)\cdot1+(\sqrt2\,i)(-\sqrt3\,i)\\
\Step{2}\;&=1+(\sqrt2-\sqrt3)i-\sqrt6\,i^2\\
\Step{3}\;&=1+(\sqrt2-\sqrt3)i+\sqrt6 \qquad (i^2=-1)\\
\Step{4}\;&=(1+\sqrt6)+(\sqrt2-\sqrt3)i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (xii)}
\textcolor{gold}{\bfseries Question:} $(\sqrt2+i)(\sqrt2-i)$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (a+b)(a-b)=a^2-b^2\\
\Step{2}\;& (\sqrt2+i)(\sqrt2-i)=(\sqrt2)^2-i^2\\
\Step{3}\;&=2-(-1)=3.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (i)}
\textcolor{gold}{\bfseries Question:} $i^9$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Powers of } i \text{ repeat every }4.\\
\Step{2}\;& 9=4\cdot 2+1 \Rightarrow i^9=i^{4\cdot2}i^1\\
\Step{3}\;&=(i^4)^2\cdot i=1^2\cdot i=i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (ii)}
\textcolor{gold}{\bfseries Question:} $i^{13}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 13=4\cdot 3+1\\
\Step{2}\;& i^{13}=i^{4\cdot3}i=(i^4)^3\cdot i=1\cdot i=i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iii)}
\textcolor{gold}{\bfseries Question:} $i^{28}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 28=4\cdot7\\
\Step{2}\;& i^{28}=(i^4)^7=1.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iv)}
\textcolor{gold}{\bfseries Question:} $(-i)^{21}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (-i)^4=1 \text{ (also repeats every 4)}\\
\Step{2}\;& 21=4\cdot5+1\\
\Step{3}\;& (-i)^{21}=((-i)^4)^5(-i)=1^5(-i)=-i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (v)}
\textcolor{gold}{\bfseries Question:} $(3i)^3$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (3i)^3=3^3\cdot i^3=27\,i^3\\
\Step{2}\;& i^3=i^2\cdot i=(-1)i=-i\\
\Step{3}\;& 27\,i^3=27(-i)=-27i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (vi)}
\textcolor{gold}{\bfseries Question:} $(-2i)^4$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (-2i)^4=(-2)^4\cdot i^4\\
\Step{2}\;&=16\cdot 1=16.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (i)}
\textcolor{gold}{\bfseries Question:} $\dfrac{3}{4-i}$ \hfill (Simplify to $a+bi$)
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{3}{4-i}\cdot\frac{4+i}{4+i}=\frac{3(4+i)}{(4-i)(4+i)}\\
\Step{2}\;&=\frac{12+3i}{4^2-(i)^2}=\frac{12+3i}{16-(-1)}=\frac{12+3i}{17}\\
\Step{3}\;&=\frac{12}{17}+\frac{3}{17}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (ii)}
\textcolor{gold}{\bfseries Question:} $\dfrac{3i}{6+5i}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{3i}{6+5i}\cdot\frac{6-5i}{6-5i}=\frac{3i(6-5i)}{(6+5i)(6-5i)}\\
\Step{2}\;&=\frac{18i-15i^2}{6^2+5^2}=\frac{18i+15}{61}\\
\Step{3}\;&=\frac{15}{61}+\frac{18}{61}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (iii)}
\textcolor{gold}{\bfseries Question:} $\dfrac{3-i\sqrt5}{3+i\sqrt5}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{3-i\sqrt5}{3+i\sqrt5}\cdot\frac{3-i\sqrt5}{3-i\sqrt5}
=\frac{(3-i\sqrt5)^2}{(3+i\sqrt5)(3-i\sqrt5)}\\
\Step{2}\;& \text{Denominator: } (3)^2-(i\sqrt5)^2=9-(-5)=14\\
\Step{3}\;& \text{Numerator: } (3-i\sqrt5)^2=9-6i\sqrt5+(i\sqrt5)^2=9-6i\sqrt5-5\\
\Step{4}\;&=\frac{4-6i\sqrt5}{14}=\frac{2}{7}-\frac{3\sqrt5}{7}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (iv)}
\textcolor{gold}{\bfseries Question:} $\dfrac{2+7i}{5i}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{2+7i}{5i}\cdot\frac{-i}{-i}=\frac{(2+7i)(-i)}{5i(-i)}\\
\Step{2}\;&=\frac{-2i-7i^2}{-5i^2}=\frac{-2i+7}{5}\\
\Step{3}\;&=\frac{7}{5}-\frac{2}{5}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (v)}
\textcolor{gold}{\bfseries Question:} $\dfrac{4+5i}{4-5i}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{4+5i}{4-5i}\cdot\frac{4+5i}{4+5i}=\frac{(4+5i)^2}{4^2+5^2}\\
\Step{2}\;& (4+5i)^2=16+40i+25i^2=16+40i-25=-9+40i\\
\Step{3}\;&=\frac{-9+40i}{41}=-\frac{9}{41}+\frac{40}{41}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (vi)}
\textcolor{gold}{\bfseries Question:} $\dfrac{3+2i}{2+i}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{3+2i}{2+i}\cdot\frac{2-i}{2-i}=\frac{(3+2i)(2-i)}{(2+i)(2-i)}\\
\Step{2}\;& \text{Denominator: }4+1=5\\
\Step{3}\;& \text{Numerator: }(3+2i)(2-i)=6-3i+4i-2i^2=6+i+2=8+i\\
\Step{4}\;&=\frac{8+i}{5}=\frac{8}{5}+\frac{1}{5}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (vii)}
\textcolor{gold}{\bfseries Question:} $\dfrac{5+i}{1+2i}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{5+i}{1+2i}\cdot\frac{1-2i}{1-2i}=\frac{(5+i)(1-2i)}{(1+2i)(1-2i)}\\
\Step{2}\;& \text{Denominator: }1+4=5\\
\Step{3}\;& \text{Numerator: }(5+i)(1-2i)=5-10i+i-2i^2=5-9i+2=7-9i\\
\Step{4}\;&=\frac{7-9i}{5}=\frac{7}{5}-\frac{9}{5}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (viii)}
\textcolor{gold}{\bfseries Question:} $\dfrac{a+ib}{a-ib}$ \hfill (assume $a,b$ real)
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{a+ib}{a-ib}\cdot\frac{a+ib}{a+ib}=\frac{(a+ib)^2}{a^2+b^2}\\
\Step{2}\;& (a+ib)^2=a^2+2aib+(ib)^2=a^2-b^2+2abi\\
\Step{3}\;& \Rightarrow \frac{a+ib}{a-ib}=\frac{a^2-b^2}{a^2+b^2}+\frac{2ab}{a^2+b^2}i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (ix)}
\textcolor{gold}{\bfseries Question:} $\dfrac{1+i}{(1-i)^2}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (1-i)^2=1-2i+i^2=1-2i-1=-2i\\
\Step{2}\;& \frac{1+i}{(1-i)^2}=\frac{1+i}{-2i}\\
\Step{3}\;& \frac{1+i}{-2i}\cdot\frac{i}{i}=\frac{(1+i)i}{-2i^2}
=\frac{i+i^2}{2}=\frac{i-1}{2}\\
\Step{4}\;&=-\frac12+\frac12 i.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (x)}
\textcolor{gold}{\bfseries Question:} $\dfrac{(2+2i)^2}{(1+i)^2}$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& (2+2i)^2=4+8i+4i^2=4+8i-4=8i\\
\Step{2}\;& (1+i)^2=1+2i+i^2=1+2i-1=2i\\
\Step{3}\;& \frac{(2+2i)^2}{(1+i)^2}=\frac{8i}{2i}=4.
\end{aligned}
\]
\end{QAPair}
