\begin{QuickBox}
{\color{cyan}\bfseries Quick rules (inequalities)}\par\medskip
\begin{itemize}
\item \textbf{Add/Subtract:} If $a<b$, then $a+c<b+c$.
\item \textbf{Multiply/Divide (positive):} If $a<b$ and $c>0$, then $ac<bc$.
\item \textbf{Multiply/Divide (negative):} If $a<b$ and $c<0$, then $ac>bc$ \textbf{(sign flips)}.
\item \textbf{Quadratic sign:} For $(x-r_1)(x-r_2)$ with $r_1<r_2$:
\[
(x-r_1)(x-r_2)>0 \Rightarrow x<r_1 \text{ or } x>r_2,\qquad
(x-r_1)(x-r_2)<0 \Rightarrow r_1<x<r_2.
\]
\item \textbf{Graph tip:} $\le,\ge$ means \textbf{closed dot}; $<,>$ means \textbf{open dot}.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} $5x-12\le 3x-4$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5x-12\le 3x-4\\
\Step{2}\;& 2x-12\le -4 \qquad (\text{subtract }3x)\\
\Step{3}\;& 2x\le 8 \qquad (\text{add }12)\\
\Step{4}\;& x\le 4 \qquad (\text{divide by }2)
\end{aligned}
\]
\[
\textbf{Solution set: }(-\infty,4].\qquad
\textbf{Graph: closed dot at }4\text{ and shade left.}
\]
\end{QAPair}

% 2
\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} $1-8x\le -4(2x-1)$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& -4(2x-1)=-8x+4\\
\Step{2}\;& 1-8x\le -8x+4\\
\Step{3}\;& 1\le 4 \qquad (\text{add }8x\text{ to both sides})
\end{aligned}
\]
Since $1\le 4$ is always true, \textbf{every real number} works.
\[
\textbf{Solution set: }(-\infty,\infty).\qquad
\textbf{Graph: shade the entire number line.}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} $-\dfrac{2}{3}x-2<\dfrac{1}{3}x+8$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& -\frac{2}{3}x-2<\frac{1}{3}x+8\\
\Step{2}\;& -x-2<8 \qquad (\text{subtract }\tfrac13x)\\
\Step{3}\;& -x<10 \qquad (\text{add }2)\\
\Step{4}\;& x>-10 \qquad (\text{multiply by }-1\text{ flips sign})
\end{aligned}
\]
\[
\textbf{Solution set: }(-10,\infty).\qquad
\textbf{Graph: open dot at }-10\text{ and shade right.}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} $8-\dfrac{4}{5}x>-14+2x$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 8-\frac45x>-14+2x\\
\Step{2}\;& 8>-14+2x+\frac45x \qquad (\text{add }\tfrac45x)\\
\Step{3}\;& 8>-14+\frac{14}{5}x\\
\Step{4}\;& 22>\frac{14}{5}x \qquad (\text{add }14)\\
\Step{5}\;& x<22\cdot\frac{5}{14}=\frac{55}{7}
\end{aligned}
\]
\[
\textbf{Solution set: }(-\infty,\tfrac{55}{7}).\qquad
\textbf{Graph: open dot at }\tfrac{55}{7}\text{ and shade left.}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} A triangle has base $(x+1)$ cm and height $8$ cm. If Area $\le 44$ sq cm, find possible $x$.
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Area}=\frac12\cdot (x+1)\cdot 8=4(x+1)\\
\Step{2}\;& 4(x+1)\le 44\\
\Step{3}\;& x+1\le 11 \qquad (\text{divide by }4)\\
\Step{4}\;& x\le 10
\end{aligned}
\]
Also, the base length must be positive: $x+1>0\Rightarrow x>-1$.
\[
\textbf{Solution set: }(-1,10].\qquad
\textbf{Graph: open dot at }-1,\text{ closed dot at }10,\text{ shade between.}
\]
\end{QAPair}

\begin{QAPair}{Question 16}
\textcolor{gold}{\bfseries Question:} $9x^2-6x+1\le 0$
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\Step{1}\; 9x^2-6x+1=(3x-1)^2
\]
A square is always $\ge 0$, so it is $\le 0$ only when it equals $0$:
\[
\Step{2}\; (3x-1)^2=0 \Rightarrow 3x-1=0 \Rightarrow x=\frac13
\]
\[
\textbf{Solution set: }\left\{\frac13\right\}.
\qquad
\textbf{Graph: a single closed dot at }\frac13.
\]
\end{QAPair}

\begin{QAPair}{Question 17}
\textcolor{gold}{\bfseries Question:} A rectangular garden has perimeter $68$ ft and area at least $240$ sq ft. Find the possible width.
\tcblower
\textcolor{green}{\bfseries Answer:}
Let width $=w$ and length $=\ell$.
\[
\Step{1}\; 2(\ell+w)=68 \Rightarrow \ell+w=34 \Rightarrow \ell=34-w
\]
\[
\Step{2}\; \text{Area}=w(34-w)\ge 240
\Rightarrow -w^2+34w-240\ge 0
\]
\[
\Step{3}\; w^2-34w+240\le 0
\Rightarrow (w-10)(w-24)\le 0
\]
Between the roots (inclusive):
\[
\textbf{Width range: } 10\le w\le 24\ \text{(feet).}
\]
\[
\textbf{Graph: closed dots at }10,24\text{ and shade between.}
\]
\end{QAPair}

% 18
\begin{QAPair}{Question 18}
\textcolor{gold}{\bfseries Question:} Are $-2,-1,0,1,2$ solutions of $2x^2+3x+1\le 0$?
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
2x^2+3x+1=(2x+1)(x+1)
\]
Zeros: $x=-1$ and $x=-\frac12$. Since the parabola opens up, $\le 0$ for
\[
-1\le x\le -\frac12
\]
Now check the given numbers:
\[
\begin{aligned}
x=-2 &: \ 2(4)+3(-2)+1=8-6+1=3>0 \ \Rightarrow \text{No}\\
x=-1 &: \ 2(1)+3(-1)+1=0 \ \Rightarrow \text{Yes}\\
x=0 &: \ 1>0 \Rightarrow \text{No}\\
x=1 &: \ 6>0 \Rightarrow \text{No}\\
x=2 &: \ 15>0 \Rightarrow \text{No}
\end{aligned}
\]
\[
\textbf{So only }x=-1\textbf{ is a solution among the given values.}
\]
\end{QAPair}

\begin{QAPair}{Question 19}
\textcolor{gold}{\bfseries Question:} $d(x)=0.05x^2+0.2x$. If maximum allowed stopping distance is $35$ m, find the speed range ($x$ in km/h) so that $d(x)\le 35$.
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 0.05x^2+0.2x\le 35\\
\Step{2}\;& 5x^2+20x\le 3500 \qquad (\times 100)\\
\Step{3}\;& x^2+4x-700\le 0 \qquad (\div 5)
\end{aligned}
\]
Roots:
\[
x=\frac{-4\pm\sqrt{16+2800}}{2}=\frac{-4\pm\sqrt{2816}}{2}
=\frac{-4\pm 16\sqrt{11}}{2}=-2\pm 8\sqrt{11}.
\]
So,
\[
\textbf{Mathematically: } \ -2-8\sqrt{11}\le x\le -2+8\sqrt{11}.
\]
Since speed cannot be negative ($x\ge 0$),
\[
\textbf{Speed range: } 0\le x\le -2+8\sqrt{11}\approx 24.53\ \text{km/h}.
\]
\end{QAPair}

% 20
\begin{QAPair}{Question 20}
\textcolor{gold}{\bfseries Question:} A rectangular garden is enclosed with a fence of total length at most $60$ m. The width is twice the length. Find the possible dimensions.
\tcblower
\textcolor{green}{\bfseries Answer:}
Let length $=L$ and width $=2L$.
\[
\Step{1}\; \text{Perimeter}=2(L+2L)=6L\le 60
\Rightarrow L\le 10
\]
Also $L>0$. Therefore
\[
\textbf{Length: }0<L\le 10\ \text{m},
\qquad
\textbf{Width: }0<2L\le 20\ \text{m}.
\]
So the possible dimensions are $(L,2L)$ with $0<L\le 10$.
\end{QAPair}
