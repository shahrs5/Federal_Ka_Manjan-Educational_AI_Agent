\begin{QuickBox}
{\color{cyan}\bfseries Quick guide: Graphing a system of linear inequalities}\par\medskip
\begin{itemize}
\item Rewrite each inequality in slope-intercept form (if possible): $y=mx+b$.
\item Draw the \textbf{boundary line}:
  \begin{itemize}
    \item \textbf{Solid line} for $\le$ or $\ge$
    \item \textbf{Dashed line} for $<$ or $>$
  \end{itemize}
\item Pick a \textbf{test point} (usually $(0,0)$ if it is not on the line).
\item Shade the half-plane that satisfies each inequality.
\item The \textbf{solution} is the \textbf{overlap (intersection)} of all shaded regions.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} Describe the steps to graph the system:
\[
x-y<6 \quad (\text{I}),\qquad y\ge 3 \quad (\text{II})
\]
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\Step{1}\; \text{Rewrite (I): } x-y<6 \Rightarrow -y<6-x \Rightarrow y>x-6.
\]
\[
\Step{2}\; \text{Draw boundary lines: }
y=x-6 \text{ (dashed)},\quad y=3 \text{ (solid)}.
\]
\[
\Step{3}\; \text{Shade: } y>x-6 \text{ means shade \emph{above} }y=x-6,\;
y\ge 3 \text{ means shade \emph{on/above} }y=3.
\]
\[
\Step{4}\; \text{The solution is the overlap of both shaded regions.}
\]

\begin{Graph}
  % Shade intersection: (y>=3) AND (y > x-6)
  \begin{scope}
    \clip (-6,3) rectangle (10,10); % y>=3
    \clip (-6,-6)--(0,-6)--(10,4)--(10,10)--(-6,10)--cycle; % y > x-6 (above line)
    \fill[cyan,opacity=0.14] (-6,-6) rectangle (10,10);
  \end{scope}

  % Boundary lines
  \draw[gold,dashed,line width=1.0pt] (0,-6)--(10,4) node[lbl,above right] {$y=x-6$ (dashed)};
  \draw[gold,line width=1.0pt] (-6,3)--(10,3) node[lbl,above right] {$y=3$ (solid)};
\end{Graph}

\textbf{Solution region:} points \textbf{above} $y=x-6$ and \textbf{on/above} $y=3$.
\end{QAPair}

\begin{QAPair}{Question 2(a)}
\textcolor{gold}{\bfseries Question:} Check whether each ordered pair is a solution of:
\[
x+y<1,\qquad x-y<2
\]
for \ $(3,1),(0,0),(-3,1),(4,1)$.
\tcblower
\textcolor{green}{\bfseries Answer:}

First rewrite $x-y<2$ as $y>x-2$.

\begin{itemize}
\item $(3,1)$: $x+y=4\not<1$ $\Rightarrow$ \textbf{Not a solution}.
\item $(0,0)$: $x+y=0<1$ and $x-y=0<2$ $\Rightarrow$ \textbf{Solution}.
\item $(-3,1)$: $x+y=-2<1$ and $x-y=-4<2$ $\Rightarrow$ \textbf{Solution}.
\item $(4,1)$: $x+y=5\not<1$ $\Rightarrow$ \textbf{Not a solution}.
\end{itemize}

\begin{Graph}
  % Shade intersection: y < 1-x AND y > x-2
  \begin{scope}
    \clip (-6,-6) rectangle (10,10);
    % y < 1-x  (below the line). In-window triangle: (-6,-6)--(7,-6)--(-6,7)
    \clip (-6,-6)--(7,-6)--(-6,7)--cycle;
    % y > x-2 (above the line). In-window polygon: (-6,-6)--(-4,-6)--(10,8)--(10,10)--(-6,10)
    \clip (-6,-6)--(-4,-6)--(10,8)--(10,10)--(-6,10)--cycle;
    \fill[cyan,opacity=0.14] (-6,-6) rectangle (10,10);
  \end{scope}

  % boundary lines
  \draw[gold,dashed,line width=1.0pt] (-6,7)--(7,-6) node[lbl,above left] {$x+y=1$ (dashed)};
  \draw[gold,dashed,line width=1.0pt] (-4,-6)--(10,8) node[lbl,above right] {$x-y=2$ (dashed)};

  % points
  \node[badpt,label={[lbl]right:$(3,1)$}] at (3,1) {};
  \node[goodpt,label={[lbl]right:$(0,0)$}] at (0,0) {};
  \node[goodpt,label={[lbl]left:$(-3,1)$}] at (-3,1) {};
  \node[badpt,label={[lbl]right:$(4,1)$}] at (4,1) {};
\end{Graph}

\end{QAPair}

\begin{QAPair}{Question 2(b)}
\textcolor{gold}{\bfseries Question:} Check whether each ordered pair is a solution of:
\[
2x-y\le 5,\qquad x+2y>2
\]
for \ $(1,-1),(4,1),(2,0),(3,2)$.
\tcblower
\textcolor{green}{\bfseries Answer:}

Rewrite:
\[
2x-y\le 5 \Rightarrow y\ge 2x-5,\qquad
x+2y>2 \Rightarrow y>1-\frac{x}{2}.
\]

\begin{itemize}
\item $(1,-1)$: $2x-y=3\le5$ (true), but $x+2y=-1\not>2$ $\Rightarrow$ \textbf{Not a solution}.
\item $(4,1)$: $2x-y=7\not\le 5$ $\Rightarrow$ \textbf{Not a solution}.
\item $(2,0)$: $2x-y=4\le5$ (true), but $x+2y=2\not>2$ $\Rightarrow$ \textbf{Not a solution}.
\item $(3,2)$: $2x-y=4\le5$ and $x+2y=7>2$ $\Rightarrow$ \textbf{Solution}.
\end{itemize}

\begin{Graph}
  % Shade intersection: y >= 2x-5 AND y > 1 - x/2
  \begin{scope}
    \clip (-6,-6) rectangle (10,10);
    % y >= 2x-5 (above line). In-window polygon:
    % line y=2x-5 hits bottom at x=-0.5, top at x=7.5, left at y=-17 (out), right at y=15 (out)
    % Safe above polygon in window: (-6,-6)--(-0.5,-6)--(7.5,10)--(-6,10)
    \clip (-6,-6)--(-0.5,-6)--(7.5,10)--(-6,10)--cycle;
    % y > 1 - x/2 (above dashed line). Line hits left: (-6,4), right:(10,-4), top y=10 at x=-18(out), bottom y=-6 at x=14(out)
    % Above polygon: (-6,4)--(10,-4)--(10,10)--(-6,10)
    \clip (-6,4)--(10,-4)--(10,10)--(-6,10)--cycle;
    \fill[cyan,opacity=0.14] (-6,-6) rectangle (10,10);
  \end{scope}

  % boundary lines
  \draw[gold,line width=1.0pt] (-0.5,-6)--(7.5,10) node[lbl,above right] {$2x-y=5$ (solid)};
  \draw[gold,dashed,line width=1.0pt] (-6,4)--(10,-4) node[lbl,above right] {$x+2y=2$ (dashed)};

  % points
  \node[badpt,label={[lbl]below:$(1,-1)$}] at (1,-1) {};
  \node[badpt,label={[lbl]right:$(4,1)$}] at (4,1) {};
  \node[badpt,label={[lbl]below:$(2,0)$}] at (2,0) {};
  \node[goodpt,label={[lbl]right:$(3,2)$}] at (3,2) {};
\end{Graph}

\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} Does the system have any solution?
\[
x-y>5,\qquad x-y<1
\]
\tcblower
\textcolor{green}{\bfseries Answer:}
Rewrite:
\[
x-y>5 \Rightarrow y<x-5,\qquad x-y<1 \Rightarrow y>x-1.
\]
A point cannot be \textbf{below} $y=x-5$ and \textbf{above} $y=x-1$ at the same time (the lines are parallel and $x-5$ is always 4 units below $x-1$).
\[
\textbf{Therefore, there is \underline{no solution}.}
\]

\begin{Graph}
  \draw[gold,dashed,line width=1.0pt] (-1,-6)--(10,5) node[lbl,above right] {$y=x-5$ (dashed)};
  \draw[gold,dashed,line width=1.0pt] (-5,-6)--(10,9) node[lbl,above right] {$y=x-1$ (dashed)};
  \node[text=redx,font=\bfseries] at (3,2) {No overlap $\Rightarrow$ no solution};
\end{Graph}
\end{QAPair}

\begin{QAPair}{Question 5 (Open-Ended)}
\textcolor{gold}{\bfseries Question:} Describe a real-world situation modeled by a system of linear inequalities. Then write and graph the system.
\tcblower
\textcolor{green}{\bfseries Answer:}

\textbf{Example situation (school supplies):}\\
Let $x$ = number of notebooks, $y$ = number of pens.
\begin{itemize}
\item Budget is \$50; notebooks cost \$3 and pens cost \$2:
\[
3x+2y\le 50
\]
\item Need at least $5$ notebooks:
\[
x\ge 5
\]
\item Want at least $10$ total items:
\[
x+y\ge 10
\]
\item Also $x\ge 0,\;y\ge 0$.
\end{itemize}

\begin{Graph}
  \begin{scope}
    \clip (0,0) rectangle (10,10); % first quadrant within our window
    % x >= 5
    \clip (5,0) rectangle (10,10);
    % x+y >= 10  (above y=10-x): polygon (0,10)--(10,0)--(10,10)--(0,10)
    \clip (0,10)--(10,0)--(10,10)--(0,10)--cycle;
    % 3x+2y <= 50 => y <= 25 - 1.5x : below that line
    % In our window, line hits y=10 at x=10, and y=0 at x=16.67(out). Use segment (10,10) to (0,25)(out)
    % Below-line region inside window includes rectangle bottom; safe clip polygon:
    \clip (0,0)--(10,0)--(10,10)--(0,10)--cycle;
    % additionally clip below the line using a large polygon under it
    \clip (-6,-20)--(30,-20)--(30,25)--(0,25)--(10,10)--cycle;
    \fill[cyan,opacity=0.14] (-6,-6) rectangle (10,10);
  \end{scope}

  % draw boundary lines
  \draw[gold,line width=1.0pt] (5,0)--(5,10) node[lbl,above right] {$x=5$};
  \draw[gold,line width=1.0pt] (0,10)--(10,0) node[lbl,above left] {$x+y=10$};
  \draw[gold,line width=1.0pt] (10,10)--(6.6667,15) node[lbl,above right] {$3x+2y=50$};

  \node[lbl] at (7,6.5) {feasible region};
\end{Graph}

\end{QAPair}
