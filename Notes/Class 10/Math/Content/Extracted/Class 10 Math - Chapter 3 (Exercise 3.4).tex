\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Cramer's rule (2 variables):}
\[
\begin{aligned}
&a_1x+b_1y=c_1,\quad a_2x+b_2y=c_2,\\
&\Delta=\begin{vmatrix}a_1&b_1\\ a_2&b_2\end{vmatrix}=a_1b_2-a_2b_1,\quad
\Delta_x=\begin{vmatrix}c_1&b_1\\ c_2&b_2\end{vmatrix},\quad
\Delta_y=\begin{vmatrix}a_1&c_1\\ a_2&c_2\end{vmatrix},\\
&x=\dfrac{\Delta_x}{\Delta},\quad y=\dfrac{\Delta_y}{\Delta}\qquad(\Delta\neq 0).
\end{aligned}
\]
\item \textbf{Inverse of a $2\times2$ matrix:}
\[
\begin{aligned}
A=\begin{pmatrix}a&b\\ c&d\end{pmatrix},\quad
A^{-1}=\frac{1}{ad-bc}\begin{pmatrix}d&-b\\ -c&a\end{pmatrix}.
\end{aligned}
\]
\item \textbf{Mixture idea:} \quad (amount)\,$\times$\,(\%) = (pure part).
\item \textbf{Perimeter of rectangle:} $P=2(L+W)$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:}
From the two acute angles of a right triangle, one angle is $10^\circ$ less than the other.
Find both acute angles by using Cramer's rule.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the angles be } x \text{ and } y. \text{ In a right triangle: } x+y=90.\\
\Step{2}\;& \text{One is }10^\circ \text{ less than the other: } x-y=-10.\\
\Step{3}\;& \text{Use Cramer's rule for }
\begin{cases}
x+y=90\\
x-y=-10
\end{cases}\\[4pt]
&\Delta=\begin{vmatrix}1&1\\ 1&-1\end{vmatrix}=(-1)-1=-2.\\
&\Delta_x=\begin{vmatrix}90&1\\ -10&-1\end{vmatrix}=90(-1)-(-10)(1)=-90+10=-80.\\
&\Delta_y=\begin{vmatrix}1&90\\ 1&-10\end{vmatrix}=1(-10)-1(90)=-100.\\
\Step{4}\;& x=\frac{\Delta_x}{\Delta}=\frac{-80}{-2}=40,\quad
y=\frac{\Delta_y}{\Delta}=\frac{-100}{-2}=50.\\
&\boxed{40^\circ \text{ and } 50^\circ.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:}
In a quiz, Abdullah correctly replied $15$ questions (out of $20$) from literature and science, getting Rs.\ $125$.
If each correct literature reply is Rs.\ $5$ and each correct science reply is Rs.\ $10$, find the correct replies of literature and science separately.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } L=\text{literature correct},\ S=\text{science correct}.\\
\Step{2}\;& L+S=15,\qquad 5L+10S=125.\\
\Step{3}\;& \Delta=\begin{vmatrix}1&1\\ 5&10\end{vmatrix}=10-5=5.\\
&\Delta_L=\begin{vmatrix}15&1\\ 125&10\end{vmatrix}=15(10)-125(1)=150-125=25.\\
&\Delta_S=\begin{vmatrix}1&15\\ 5&125\end{vmatrix}=1(125)-5(15)=125-75=50.\\
\Step{4}\;& L=\frac{\Delta_L}{\Delta}=\frac{25}{5}=5,\quad
S=\frac{\Delta_S}{\Delta}=\frac{50}{5}=10.\\
&\boxed{L=5,\ S=10.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:}
If Asad bought $3$ notebooks and $2$ books paying Rs.\ $600$ and Iman bought $4$ notebooks and $1$ book paying Rs.\ $550$,
find the cost of each notebook and book. (Use Cramer's rule.)\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } n=\text{cost of a notebook},\ b=\text{cost of a book}.\\
\Step{2}\;& 3n+2b=600,\qquad 4n+b=550.\\
\Step{3}\;& \Delta=\begin{vmatrix}3&2\\ 4&1\end{vmatrix}=3(1)-4(2)=3-8=-5.\\
&\Delta_n=\begin{vmatrix}600&2\\ 550&1\end{vmatrix}=600(1)-550(2)=600-1100=-500.\\
&\Delta_b=\begin{vmatrix}3&600\\ 4&550\end{vmatrix}=3(550)-4(600)=1650-2400=-750.\\
\Step{4}\;& n=\frac{\Delta_n}{\Delta}=\frac{-500}{-5}=100,\quad
b=\frac{\Delta_b}{\Delta}=\frac{-750}{-5}=150.\\
&\boxed{\text{Notebook }= \text{Rs.\ }100,\ \text{Book }= \text{Rs.\ }150.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:}
Sum of ages of two sisters Sundas and Zenab is $22$ years.
If Sundas is $1$ year older than twice the age of Zenab, find their ages using matrix inversion method.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } S=\text{age of Sundas},\ Z=\text{age of Zenab}.\\
\Step{2}\;& S+Z=22,\qquad S-2Z=1.\\
\Step{3}\;& \text{Write } A\mathbf{x}=\mathbf{c}:
\quad
\begin{pmatrix}1&1\\ 1&-2\end{pmatrix}
\begin{pmatrix}S\\ Z\end{pmatrix}
=
\begin{pmatrix}22\\ 1\end{pmatrix}.\\
\Step{4}\;& \det(A)=1(-2)-1(1)=-3\neq 0,\\
& A^{-1}=\frac{1}{-3}\begin{pmatrix}-2&-1\\ -1&1\end{pmatrix}.\\
\Step{5}\;&
\begin{pmatrix}S\\ Z\end{pmatrix}
=A^{-1}\begin{pmatrix}22\\ 1\end{pmatrix}
=\frac{1}{-3}\begin{pmatrix}-2&-1\\ -1&1\end{pmatrix}\begin{pmatrix}22\\ 1\end{pmatrix}
=\frac{1}{-3}\begin{pmatrix}-45\\ -21\end{pmatrix}
=\begin{pmatrix}15\\ 7\end{pmatrix}.\\
&\boxed{S=15\text{ years},\ Z=7\text{ years}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:}
Two trucks have capacities of $10$ tons and $12$ tons. They made a total of $20$ round trips to haul $226$ tons of sand.
How many round trips did each truck make?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } x=\text{trips by 10-ton truck},\ y=\text{trips by 12-ton truck}.\\
\Step{2}\;& x+y=20,\qquad 10x+12y=226.\\
\Step{3}\;& x=20-y \Rightarrow 10(20-y)+12y=226\\
&\Rightarrow 200-10y+12y=226 \Rightarrow 2y=26 \Rightarrow y=13.\\
\Step{4}\;& x=20-13=7.\\
&\boxed{\text{10-ton truck: }7\text{ trips, \ \ 12-ton truck: }13\text{ trips.}}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:}
To make $20$ kg of aluminum alloy with $70\%$ aluminum, a metallurgist wants to use two metals with $55\%$ and $80\%$ aluminum.
How much of each metal should he use?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } x=\text{kg of 55\% metal},\ y=\text{kg of 80\% metal}.\\
\Step{2}\;& x+y=20.\\
\Step{3}\;& 0.55x+0.80y=0.70(20)=14.\\
\Step{4}\;& x=20-y \Rightarrow 0.55(20-y)+0.80y=14\\
&\Rightarrow 11-0.55y+0.80y=14 \Rightarrow 0.25y=3 \Rightarrow y=12.\\
\Step{5}\;& x=20-12=8.\\
&\boxed{55\%\text{ metal: }8\text{ kg,}\quad 80\%\text{ metal: }12\text{ kg.}}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:}
Soybean meal contains $16\%$ protein and corn meal contains $9\%$ protein.
How many pounds of each should be mixed to get a $350$-lb mixture that is $12\%$ protein?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } x=\text{lb soybean (16\%)},\ y=\text{lb corn (9\%)}.\\
\Step{2}\;& x+y=350.\\
\Step{3}\;& 0.16x+0.09y=0.12(350)=42.\\
\Step{4}\;& x=350-y \Rightarrow 0.16(350-y)+0.09y=42\\
&\Rightarrow 56-0.16y+0.09y=42 \Rightarrow 56-0.07y=42 \Rightarrow 0.07y=14\\
&\Rightarrow y=200.\\
\Step{5}\;& x=350-200=150.\\
&\boxed{\text{Soybean: }150\text{ lb, \ \ Corn: }200\text{ lb.}}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:}
The perimeter of a lot is $190$ m. The width is one fourth of the length. Find the dimensions.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let } L=\text{length},\ W=\text{width}. \quad 2(L+W)=190 \Rightarrow L+W=95.\\
\Step{2}\;& W=\frac14 L.\\
\Step{3}\;& L+\frac14L=95 \Rightarrow \frac54L=95 \Rightarrow L=95\cdot\frac{4}{5}=76.\\
\Step{4}\;& W=\frac14(76)=19.\\
&\boxed{L=76\text{ m},\ W=19\text{ m.}}
\end{aligned}
\]
\end{QAPair}
