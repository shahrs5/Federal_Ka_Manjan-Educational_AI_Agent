\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (absolute value)}\par\medskip
\begin{itemize}
\item \textbf{Equation:} $|A|=k\ (k\ge 0)\ \Longrightarrow\ A=k\ \text{or}\ A=-k$.
\item \textbf{No solution:} $|A|=k$ with $k<0$ has \textbf{no solution}.
\item \textbf{Inequality (inside form):}
\[
|A|\le k \Longleftrightarrow -k\le A\le k,\qquad
|A|<k \Longleftrightarrow -k< A< k.
\]
\[
|A|\ge k \Longleftrightarrow A\ge k\ \text{or}\ A\le -k,\qquad
|A|>k \Longleftrightarrow A>k\ \text{or}\ A<-k.
\]
\item \textbf{Absolute deviation:} $|x-a|\le d \Longleftrightarrow a-d\le x\le a+d$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} $8|x-3|=88$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 8|x-3|=88 \;\Rightarrow\; |x-3|=11.\\
\Step{2}\;& x-3=11 \ \text{or}\ x-3=-11.\\
\Step{3}\;& x=14 \ \text{or}\ x=-8.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} $|2x+9|=30$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 2x+9=30 \ \text{or}\ 2x+9=-30.\\
\Step{2}\;& 2x=21 \ \text{or}\ 2x=-39.\\
\Step{3}\;& x=\frac{21}{2} \ \text{or}\ x=-\frac{39}{2}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} $|x-3|=-26$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |x-3|\ge 0 \ \text{for all real }x.\\
\Step{2}\;& \text{But } -26<0.\\
\Step{3}\;& \Rightarrow \textbf{No solution.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} $3|x+6|=9x-6$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 3|x+6|=9x-6 \;\Rightarrow\; |x+6|=3x-2.\\
\Step{2}\;& \text{Since }|x+6|\ge 0,\ \text{we must have }3x-2\ge 0 \Rightarrow x\ge \frac{2}{3}.\\
\Step{3}\;& x+6=3x-2 \ \text{or}\ -(x+6)=3x-2.\\
\Step{4}\;& x+6=3x-2 \Rightarrow 8=2x \Rightarrow x=4.\\
\Step{5}\;& -(x+6)=3x-2 \Rightarrow -x-6=3x-2 \Rightarrow -4x=4 \Rightarrow x=-1.\\
\Step{6}\;& x=-1 \text{ does not satisfy } x\ge \frac{2}{3}.\\
\Step{7}\;& \Rightarrow \boxed{x=4}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} $|x-3|=2x$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |x-3|=2x \Rightarrow 2x\ge 0 \Rightarrow x\ge 0.\\
\Step{2}\;& x-3=2x \ \text{or}\ -(x-3)=2x.\\
\Step{3}\;& x-3=2x \Rightarrow x=-3 \ (\text{reject since }x\ge 0).\\
\Step{4}\;& -x+3=2x \Rightarrow 3=3x \Rightarrow x=1.\\
\Step{5}\;& \Rightarrow \boxed{x=1}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} $\left|\dfrac{4-5x}{6}\right|=7$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \left|\frac{4-5x}{6}\right|=7 \Rightarrow |4-5x|=42.\\
\Step{2}\;& 4-5x=42 \ \text{or}\ 4-5x=-42.\\
\Step{3}\;& -5x=38 \Rightarrow x=-\frac{38}{5},\qquad -5x=-46 \Rightarrow x=\frac{46}{5}.\\
\Step{4}\;& \Rightarrow \boxed{x=-\frac{38}{5}\ \text{or}\ x=\frac{46}{5}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:} Explain why $|3x-6|+7=4$ has no solution.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |3x-6|\ge 0 \ \text{for all }x.\\
\Step{2}\;& \Rightarrow |3x-6|+7 \ge 7.\\
\Step{3}\;& \text{But the equation says }|3x-6|+7=4,\ \text{and }4<7.\\
\Step{4}\;& \Rightarrow \textbf{No solution.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:} A basketball is inflated to $8$ psi. (Absolute error is the absolute deviation.) Find the minimum and maximum acceptable air pressure.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let }p\text{ be the actual pressure (psi), and let }E\text{ be the maximum allowed absolute error.}\\
\Step{2}\;& \text{``Absolute deviation from 8 is at most }E\text{'' means } |p-8|\le E.\\
\Step{3}\;& |p-8|\le E \Longleftrightarrow 8-E \le p \le 8+E.\\
\Step{4}\;& \Rightarrow \textbf{Minimum pressure }=8-E,\quad \textbf{Maximum pressure }=8+E.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:} $|9x-1|\le 10$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |9x-1|\le 10 \Longleftrightarrow -10 \le 9x-1 \le 10.\\
\Step{2}\;& -9 \le 9x \le 11.\\
\Step{3}\;& -1 \le x \le \frac{11}{9}.\\
\Step{4}\;& \textbf{Interval: } \boxed{\left[-1,\ \frac{11}{9}\right]}\ \ (\text{closed dots, shade between}).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} $|2x-7|<1$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |2x-7|<1 \Longleftrightarrow -1<2x-7<1.\\
\Step{2}\;& 6<2x<8.\\
\Step{3}\;& 3<x<4.\\
\Step{4}\;& \textbf{Interval: } \boxed{(3,4)}\ \ (\text{open dots, shade between}).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 11}
\textcolor{gold}{\bfseries Question:} $5\left|\dfrac{1}{2}x+3\right|>5$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5\left|\frac12 x+3\right|>5 \Rightarrow \left|\frac12 x+3\right|>1.\\
\Step{2}\;& \frac12 x+3>1 \ \text{or}\ \frac12 x+3<-1.\\
\Step{3}\;& \frac12 x>-2 \Rightarrow x>-4,\qquad \frac12 x<-4 \Rightarrow x<-8.\\
\Step{4}\;& \textbf{Solution: } \boxed{(-\infty,-8)\ \cup\ (-4,\infty)}\ \ (\text{open dots at }-8,-4).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 12}
\textcolor{gold}{\bfseries Question:} $3|14-x|>6$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 3|14-x|>6 \Rightarrow |14-x|>2.\\
\Step{2}\;& |x-14|>2 \Rightarrow x-14>2 \ \text{or}\ x-14<-2.\\
\Step{3}\;& x>16 \ \text{or}\ x<12.\\
\Step{4}\;& \textbf{Solution: } \boxed{(-\infty,12)\ \cup\ (16,\infty)}\ \ (\text{open dots at }12,16).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 13}
\textcolor{gold}{\bfseries Question:} $\left|\dfrac{3x-2}{5}\right|\le 2$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \left|\frac{3x-2}{5}\right|\le 2 \Rightarrow |3x-2|\le 10.\\
\Step{2}\;& -10 \le 3x-2 \le 10.\\
\Step{3}\;& -8 \le 3x \le 12.\\
\Step{4}\;& -\frac{8}{3}\le x \le 4.\\
\Step{5}\;& \textbf{Interval: } \boxed{\left[-\frac{8}{3},\ 4\right]}\ \ (\text{closed dots, shade between}).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 14}
\textcolor{gold}{\bfseries Question:} $\left|\dfrac{2-5x}{4}\right|\ge \dfrac{2}{3}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \left|\frac{2-5x}{4}\right|\ge \frac{2}{3}
\Rightarrow |2-5x|\ge \frac{8}{3}.\\
\Step{2}\;& 2-5x \ge \frac{8}{3}\ \text{or}\ 2-5x \le -\frac{8}{3}.\\
\Step{3}\;& 2=\frac{6}{3}: \ \frac{6}{3}-5x \ge \frac{8}{3} \Rightarrow -5x \ge \frac{2}{3} \Rightarrow x\le -\frac{2}{15}.\\
\Step{4}\;& \frac{6}{3}-5x \le -\frac{8}{3} \Rightarrow -5x \le -\frac{14}{3} \Rightarrow x\ge \frac{14}{15}.\\
\Step{5}\;& \textbf{Solution: } \boxed{(-\infty,-\frac{2}{15}]\ \cup\ [\frac{14}{15},\infty)}\ \ (\text{closed endpoints}).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 15}
\textcolor{gold}{\bfseries Question:} An essay must be 500 words with an absolute deviation of at most 30 words. What are the possible numbers of words?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let }w\text{ be the number of words. ``At most 30 away from 500'' means }|w-500|\le 30.\\
\Step{2}\;& |w-500|\le 30 \Longleftrightarrow 500-30 \le w \le 500+30.\\
\Step{3}\;& \Rightarrow \boxed{470 \le w \le 530}.\\
\Step{4}\;& \text{(If }w\text{ must be an integer, then }w=470,471,\dots,530\text{.)}
\end{aligned}
\]
\end{QAPair}
