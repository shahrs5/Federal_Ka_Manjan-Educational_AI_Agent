\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Consecutive even numbers:} $2n,\;2n+2$.
\item \textbf{Consecutive odd numbers:} $2n+1,\;2n+3$.
\item \textbf{Two-digit number:} $10a+b$ (tens digit $a$, units digit $b$), reversed $10b+a$.
\item \textbf{If } x+y=s,\;xy=p,\text{ then } t^2-st+p=0 \text{ has roots } x,y.
\item \textbf{Identity:} $(x+y)^2=x^2+y^2+2xy$.
\item \textbf{Reciprocal-sum:} $\displaystyle \frac{1}{\frac1x+\frac1y}=\frac{xy}{x+y}$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} Product of two consecutive positive even numbers is $120$. Find the numbers.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the numbers be } 2n \text{ and } 2n+2,\; n\in\mathbb{N}.\\
\Step{2}\;& (2n)(2n+2)=120 \;\Rightarrow\; 4n(n+1)=120 \;\Rightarrow\; n(n+1)=30.\\
\Step{3}\;& n^2+n-30=0 \;\Rightarrow\; (n-5)(n+6)=0.\\
\Step{4}\;& n=5 \;(\text{positive}) \Rightarrow 2n=10,\; 2n+2=12.\\
&\boxed{10 \text{ and } 12.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} The difference of a positive number and its square is $380$. Find the number.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the number be } x>0. \text{ For a positive number, } x^2>x,\\
&\text{so we take } x^2-x=380.\\
\Step{2}\;& x^2-x-380=0.\\
\Step{3}\;& x=\frac{1\pm\sqrt{1+1520}}{2}=\frac{1\pm\sqrt{1521}}{2}
=\frac{1\pm 39}{2}.\\
\Step{4}\;& x=20 \text{ or } x=-19,\ \text{but } x>0.\\
&\boxed{x=20.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} The difference of cubes of two consecutive positive numbers is $91$. Find them.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the numbers be } n \text{ and } n+1,\; n\in\mathbb{N}.\\
\Step{2}\;& (n+1)^3-n^3=91.\\
\Step{3}\;& 3n^2+3n+1=91 \;\Rightarrow\; 3n^2+3n-90=0.\\
\Step{4}\;& n^2+n-30=0 \;\Rightarrow\; (n-5)(n+6)=0 \Rightarrow n=5.\\
&\boxed{5 \text{ and } 6.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} The sum of the Cartesian coordinates of a point is $9$ and the sum of their squares is $45$. Find the coordinates.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the point be }(x,y). \text{ Given } x+y=9,\; x^2+y^2=45.\\
\Step{2}\;& (x+y)^2=x^2+y^2+2xy \Rightarrow 9^2=45+2xy.\\
\Step{3}\;& 81=45+2xy \Rightarrow 2xy=36 \Rightarrow xy=18.\\
\Step{4}\;& \text{Now } x,y \text{ are roots of } t^2-9t+18=0.\\
\Step{5}\;& (t-3)(t-6)=0 \Rightarrow t=3 \text{ or } 6.\\
&\boxed{(x,y)=(3,6)\ \text{or}\ (6,3).}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} The sum of two numbers is $11$ and the product is $30$. Find the numbers.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the numbers be } x \text{ and } y. \quad x+y=11,\; xy=30.\\
\Step{2}\;& \text{Then } t^2-11t+30=0 \text{ has roots } x,y.\\
\Step{3}\;& (t-5)(t-6)=0 \Rightarrow t=5 \text{ or } 6.\\
&\boxed{5 \text{ and } 6.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} The sum of the squares of two consecutive positive odd integers is $34$. Find the integers.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the odd integers be } 2n+1 \text{ and } 2n+3,\; n\ge 0.\\
\Step{2}\;& (2n+1)^2+(2n+3)^2=34.\\
\Step{3}\;& (4n^2+4n+1)+(4n^2+12n+9)=34\\
&\Rightarrow 8n^2+16n+10=34 \Rightarrow 8n^2+16n-24=0.\\
\Step{4}\;& n^2+2n-3=0 \Rightarrow (n-1)(n+3)=0 \Rightarrow n=1.\\
\Step{5}\;& 2n+1=3,\; 2n+3=5.\\
&\boxed{3 \text{ and } 5.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:} The sum of ages of a father and his son is $50$ years. Ten years ago, the father was $9$ times as old as his son. Find their present ages.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let present ages be } F \text{ (father) and } S \text{ (son).}\\
\Step{2}\;& F+S=50.\\
\Step{3}\;& \text{Ten years ago: } F-10=9(S-10).\\
\Step{4}\;& F-10=9S-90 \Rightarrow F=9S-80.\\
\Step{5}\;& (9S-80)+S=50 \Rightarrow 10S=130 \Rightarrow S=13.\\
\Step{6}\;& F=50-13=37.\\
&\boxed{\text{Father }=37 \text{ years, Son }=13 \text{ years.}}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:} A two-digit number is decreased by $45$ when the digits are reversed. If the sum of the digits is $11$, find the number.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the tens digit be } a \ (1\le a\le 9), \text{ units digit } b \ (0\le b\le 9).\\
\Step{2}\;& \text{Number }=10a+b,\ \text{reversed }=10b+a.\\
\Step{3}\;& \text{Reversing decreases the number by }45:\ (10a+b)-(10b+a)=45.\\
\Step{4}\;& 9a-9b=45 \Rightarrow a-b=5.\\
\Step{5}\;& \text{Also } a+b=11. \Rightarrow
\begin{cases}
a-b=5\\
a+b=11
\end{cases}
\Rightarrow 2a=16 \Rightarrow a=8,\ b=3.\\
\Step{6}\;& \text{Number }=10(8)+3=83.\\
&\boxed{83.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:} Sum of two numbers is $20$. Find the numbers if the sum of first number and square of the other is $40$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let first number be } x,\ \text{other be } y.\\
\Step{2}\;& x+y=20,\quad x+y^2=40.\\
\Step{3}\;& x=20-y. \text{ Substitute into } x+y^2=40:\\
& (20-y)+y^2=40 \Rightarrow y^2-y-20=0.\\
\Step{4}\;& (y-5)(y+4)=0 \Rightarrow y=5 \text{ or } y=-4.\\
\Step{5}\;& x=20-y \Rightarrow x=15 \text{ or } x=24.\\
&\boxed{(x,y)=(15,5)\ \text{or}\ (24,-4).}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} The reciprocal of the sum of reciprocals of two numbers is $\frac{12}{5}$. Find the numbers if their sum is $10$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the numbers be } x \text{ and } y,\ \ x\neq 0,\ y\neq 0.\\
\Step{2}\;& \frac{1}{\frac1x+\frac1y}=\frac{12}{5}.\\
\Step{3}\;& \frac{1}{\frac{x+y}{xy}}=\frac{xy}{x+y}=\frac{12}{5}.\\
\Step{4}\;& x+y=10 \Rightarrow \frac{xy}{10}=\frac{12}{5} \Rightarrow xy=24.\\
\Step{5}\;& \text{Now } t^2-10t+24=0 \text{ has roots } x,y.\\
\Step{6}\;& (t-4)(t-6)=0 \Rightarrow t=4 \text{ or } 6.\\
&\boxed{4 \text{ and } 6.}
\end{aligned}
\]
\end{QAPair}
