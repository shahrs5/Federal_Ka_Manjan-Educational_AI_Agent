\begin{QuickBox}
{\color{cyan}\bfseries Quick methods (Simultaneous equations)}\par\medskip
\begin{itemize}
\item \textbf{Substitution:} Solve one equation for $y$ (or $x$) and substitute into the other.
\item \textbf{Elimination:} Add/subtract equations to eliminate a variable.
\item \textbf{Useful factoring:} $a^2-b^2=(a-b)(a+b)$.
\item \textbf{When only squares appear:} Let $A=x^2,\;B=y^2$ to make a linear system in $A,B$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} $2x+y=1;\quad x^2+y^2=10$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&2x+y=1 \Rightarrow y=1-2x.\\
\Step{2}\;&x^2+(1-2x)^2=10\\
&\Rightarrow x^2+(1-4x+4x^2)=10\\
&\Rightarrow 5x^2-4x-9=0.\\
\Step{3}\;&x=\frac{4\pm\sqrt{16+180}}{10}=\frac{4\pm 14}{10}
\Rightarrow x=\frac95 \text{ or } x=-1.\\
\Step{4}\;&y=1-2x \Rightarrow
\begin{cases}
x=\frac95 \Rightarrow y=-\frac{13}{5},\\
x=-1 \Rightarrow y=3.
\end{cases}
\end{aligned}
\]
\[
\boxed{(x,y)=\left(\frac95,-\frac{13}{5}\right),\;(-1,3).}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} $3x-2y=1;\quad x^2+xy-y^2=1$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&3x-2y=1 \Rightarrow y=\frac{3x-1}{2}.\\
\Step{2}\;&x^2+x\!\left(\frac{3x-1}{2}\right)-\left(\frac{3x-1}{2}\right)^2=1\\
&\Rightarrow (x-1)(x+5)=0.\\
\Step{3}\;&x=1 \text{ or } x=-5.\\
\Step{4}\;&y=\frac{3x-1}{2}\Rightarrow
\begin{cases}
x=1 \Rightarrow y=1,\\
x=-5 \Rightarrow y=-8.
\end{cases}
\end{aligned}
\]
\[
\boxed{(x,y)=(1,1),\;(-5,-8).}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} $3x+y+3=0;\quad (x+1)^2-4(x+1)-6=y$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&3x+y+3=0 \Rightarrow y=-3x-3.\\
\Step{2}\;&(x+1)^2-4(x+1)-6
= x^2+2x+1-4x-4-6
= x^2-2x-9.\\
\Step{3}\;&x^2-2x-9 = -3x-3
\Rightarrow x^2+x-6=0
\Rightarrow (x+3)(x-2)=0.\\
\Step{4}\;&x=-3 \text{ or } x=2.\\
\Step{5}\;&y=-3x-3 \Rightarrow
\begin{cases}
x=-3 \Rightarrow y=6,\\
x=2 \Rightarrow y=-9.
\end{cases}
\end{aligned}
\]
\[
\boxed{(x,y)=(-3,6),\;(2,-9).}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} $(x+3)^2-(y-2)^2=10;\quad x+y=4$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&x+y=4 \Rightarrow y=4-x.\\
\Step{2}\;&(x+3)^2-\bigl((4-x)-2\bigr)^2=10\\
&\Rightarrow (x+3)^2-(2-x)^2=10\\
&\Rightarrow \bigl((x+3)-(x-2)\bigr)\bigl((x+3)+(x-2)\bigr)=10\\
&\Rightarrow (5)(2x+1)=10\\
&\Rightarrow 2x+1=2 \Rightarrow x=\frac12.\\
\Step{3}\;&y=4-\frac12=\frac72.
\end{aligned}
\]
\[
\boxed{(x,y)=\left(\frac12,\frac72\right).}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} $2x^2-8xy+6y^2=0;\quad x^2+y^2=45$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&2x^2-8xy+6y^2=2(x^2-4xy+3y^2)=2(x-y)(x-3y)=0.\\
\Step{2}\;&\Rightarrow x=y \;\text{or}\; x=3y.
\end{aligned}
\]

\textcolor{muted}{\textbf{Case 1:}} $x=y$
\[
\begin{aligned}
x^2+y^2=45 \Rightarrow 2x^2=45 \Rightarrow x^2=\frac{45}{2}
\Rightarrow x=\pm \frac{3\sqrt{10}}{2},\; y=x.
\end{aligned}
\]

\textcolor{muted}{\textbf{Case 2:}} $x=3y$
\[
\begin{aligned}
x^2+y^2=45 \Rightarrow 9y^2+y^2=45 \Rightarrow 10y^2=45
\Rightarrow y^2=\frac{9}{2}
\Rightarrow y=\pm \frac{3\sqrt2}{2},\; x=3y=\pm\frac{9\sqrt2}{2}.
\end{aligned}
\]

\[
\boxed{
(x,y)=\left(\pm\frac{3\sqrt{10}}{2},\pm\frac{3\sqrt{10}}{2}\right),\;
\left(\pm\frac{9\sqrt2}{2},\pm\frac{3\sqrt2}{2}\right)
\text{ (same sign in each pair).}
}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} $6x^2-5xy-y^2=0;\quad y^2+4xy=30$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&6x^2-5xy-y^2=(6x+y)(x-y)=0.\\
\Step{2}\;&\Rightarrow 6x+y=0 \;\text{or}\; x-y=0.
\end{aligned}
\]

\textcolor{muted}{\textbf{Case 1:}} $x=y$
\[
\begin{aligned}
y^2+4xy=30 \Rightarrow y^2+4y^2=30 \Rightarrow 5y^2=30
\Rightarrow y^2=6 \Rightarrow y=\pm\sqrt6,\; x=y.
\end{aligned}
\]

\textcolor{muted}{\textbf{Case 2:}} $y=-6x$
\[
\begin{aligned}
y^2+4xy=30
&\Rightarrow 36x^2+4x(-6x)=30\\
&\Rightarrow 36x^2-24x^2=30\\
&\Rightarrow 12x^2=30 \Rightarrow x^2=\frac52 \Rightarrow x=\pm\frac{\sqrt{10}}{2},\\
&\Rightarrow y=-6x=\mp 3\sqrt{10}.
\end{aligned}
\]

\[
\boxed{
(x,y)=(\sqrt6,\sqrt6),\;(-\sqrt6,-\sqrt6),\;
\left(\frac{\sqrt{10}}{2},-3\sqrt{10}\right),\;
\left(-\frac{\sqrt{10}}{2},3\sqrt{10}\right).
}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:} $4x^2+4y^2=65;\quad 6x-2y=5$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&4x^2+4y^2=65 \Rightarrow x^2+y^2=\frac{65}{4}.\\
\Step{2}\;&6x-2y=5 \Rightarrow 3x-y=\frac52 \Rightarrow y=3x-\frac52.\\
\Step{3}\;&x^2+\left(3x-\frac52\right)^2=\frac{65}{4}\\
&\Rightarrow x^2+\left(9x^2-15x+\frac{25}{4}\right)=\frac{65}{4}\\
&\Rightarrow 10x^2-15x-10=0\\
&\Rightarrow 2x^2-3x-2=0\\
&\Rightarrow (2x+1)(x-2)=0.
\end{aligned}
\]
\[
\begin{aligned}
\Step{4}\;&x=2 \text{ or } x=-\frac12.\\
\Step{5}\;&y=3x-\frac52 \Rightarrow
\begin{cases}
x=2 \Rightarrow y=\frac72,\\
x=-\frac12 \Rightarrow y=-4.
\end{cases}
\end{aligned}
\]
\[
\boxed{(x,y)=\left(2,\frac72\right),\left(-\frac12,-4\right).}
\]
\end{QAPair}

\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:} $x^2+y=0;\quad x+y=-2$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&x^2+y=0 \Rightarrow y=-x^2.\\
\Step{2}\;&x+(-x^2)=-2 \Rightarrow -x^2+x+2=0\\
&\Rightarrow x^2-x-2=0 \Rightarrow (x-2)(x+1)=0.\\
\Step{3}\;&x=2 \text{ or } x=-1.\\
\Step{4}\;&y=-x^2 \Rightarrow
\begin{cases}
x=2 \Rightarrow y=-4,\\
x=-1 \Rightarrow y=-1.
\end{cases}
\end{aligned}
\]
\[
\boxed{(x,y)=(2,-4),\;(-1,-1).}
\]
\end{QAPair}

\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:} $x^2+3y^2=14;\quad 3x^2+y^2=6$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
Let $A=x^2$ and $B=y^2$. Then
\[
\begin{aligned}
\Step{1}\;&A+3B=14,\quad 3A+B=6.\\
\Step{2}\;&3(A+3B)=42 \Rightarrow 3A+9B=42.\\
\Step{3}\;&(3A+9B)-(3A+B)=42-6 \Rightarrow 8B=36 \Rightarrow B=\frac92.\\
\Step{4}\;&3A+B=6 \Rightarrow 3A=6-\frac92=\frac32 \Rightarrow A=\frac12.
\end{aligned}
\]
So
\[
x=\pm\sqrt{\frac12}=\pm\frac{\sqrt2}{2},\qquad
y=\pm\sqrt{\frac92}=\pm\frac{3\sqrt2}{2}.
\]
\[
\boxed{(x,y)=\left(\pm\frac{\sqrt2}{2},\,\pm\frac{3\sqrt2}{2}\right)\ \text{(all sign combinations)}.}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} $2x^2-5y^2=8;\quad x^2+2y^2=13$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
Let $A=x^2$ and $B=y^2$. Then
\[
\begin{aligned}
\Step{1}\;&2A-5B=8,\quad A+2B=13.\\
\Step{2}\;&A=13-2B.\\
\Step{3}\;&2(13-2B)-5B=8 \Rightarrow 26-9B=8 \Rightarrow 9B=18 \Rightarrow B=2.\\
\Step{4}\;&A=13-2(2)=9.
\end{aligned}
\]
So
\[
x=\pm 3,\qquad y=\pm \sqrt2.
\]
\[
\boxed{(x,y)=(\pm 3,\pm \sqrt2)\ \text{(all sign combinations)}.}
\]
\end{QAPair}
