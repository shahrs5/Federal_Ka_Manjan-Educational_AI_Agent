\begin{QuickBox}

\begin{itemize}

\item $a^3+b^3=(a+b)(a^2-ab+b^2)$

\end{itemize}

\begin{QAPair}{Question 1 (i) --- MCQ}
\textcolor{gold}{\bfseries Question:} Factors of $-2-a+a^2$ are:
\begin{itemize}
\item[(a)] $(a-2)(a-1)$
\item[(b)] $(a+1)(a+2)$
\item[(c)] $(a+2)(a-1)$
\item[(d)] $(a+1)(a-2)$
\end{itemize}
\tcblower
\textcolor{green}{\bfseries Answer:} \textbf{(d)}
\[
\begin{aligned}
\Step{1}\;& -2-a+a^2=a^2-a-2.\\
\Step{2}\;& a^2-a-2=(a-2)(a+1).\\
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ii) --- MCQ}

\begin{QAPair}{Question 1 (xvi) --- MCQ}
\textcolor{gold}{\bfseries Question:} If product of two polynomials is $(a-b)^2(a^2+ab+b^2)$ and their HCF is $a-b$, what is their LCM?
\begin{itemize}
\item[(a)] $a^3-b^3$
\item[(b)] $(a-b)^3(a^2+ab+b^2)$
\item[(c)] $(a-b)^2(a^2+ab+b^2)$
\item[(d)] $a^2+ab+b^2$
\end{itemize}
\tcblower
\textcolor{green}{\bfseries Answer:} \textbf{(a)}
\[
\text{LCM}=\frac{\text{product}}{\text{HCF}}
=\frac{(a-b)^2(a^2+ab+b^2)}{(a-b)}
=(a-b)(a^2+ab+b^2)=a^3-b^3.
\]
\end{QAPair}

Find HCF and LCM of:

a(a+c)-b(b+c),\quad b(b+a)-c(c+a),\quad c(c+b)-a(a+b).
\]
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
a(a+c)-b(b+c)=(a-b)(a+b+c),
\]
\[
b(b+a)-c(c+a)=(b-c)(a+b+c),
\]
\[
c(c+b)-a(a+b)=(c-a)(a+b+c).
\]
\[
\boxed{\text{HCF}=a+b+c}
\]
\[
\boxed{\text{LCM}=(a+b+c)(a-b)(b-c)(c-a)}
\]
\end{QAPair}

\]
\end{QAPair}

\begin{QAPair}{Question 20}
Find $a$ and $b$ if
\[
x^4+ax^3+bx^2-4x+4
\]
is a perfect square.
\tcblower
\textcolor{green}{\bfseries Answer:}
Assume $(x^2+px+q)^2=x^4+2px^3+(p^2+2q)x^2+2pq\,x+q^2$.

Match coefficients:
\[
2pq=-4\Rightarrow pq=-2,\qquad q^2=4\Rightarrow q=\pm2.
\]
If $q=2$, then $p=-1$ $\Rightarrow a=2p=-2,\ b=p^2+2q=1+4=5$.

If $q=-2$, then $p=1$ $\Rightarrow a=2,\ b=1-4=-3$.

\[
\boxed{(a,b)=(-2,5)\ \text{or}\ (2,-3)}
\]
\end{QAPair}
