\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Reciprocal:} If $A\neq 0$, then $\dfrac{1}{A}$ is the reciprocal of $A$.
\item \textbf{Division of fractions:} $\displaystyle \frac{A}{B}\div \frac{C}{D}=\frac{A}{B}\times \frac{D}{C}$ \; (all denominators nonzero).
\item \textbf{Cancel common factors:} Factor first, then cancel identical factors in numerator/denominator.
\item \textbf{Identities:} $a^2-b^2=(a-b)(a+b)$,\;\; $a^4-b^4=(a-b)(a+b)(a^2+b^2)$.
\item \textbf{Add/Subtract fractions:} take LCM of denominators (common denominator) and combine numerators.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1(i)}
\textcolor{gold}{\bfseries Question:} Product of what algebraic fraction and $x^3+7x-8$ is $1$?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{A number times its reciprocal equals }1.\\
\Step{2}\;& \boxed{\text{Required fraction }=\dfrac{1}{x^3+7x-8}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1(ii)}
\textcolor{gold}{\bfseries Question:} Which algebraic fraction divided by $\dfrac{x^2}{x^2+y^2}$ gives $1$?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{If }F \div \frac{x^2}{x^2+y^2}=1,\text{ then }F=\frac{x^2}{x^2+y^2}.\\
\Step{2}\;& \boxed{\text{Required fraction }=\dfrac{x^2}{x^2+y^2}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1(iii)}
\textcolor{gold}{\bfseries Question:} Sum of what algebraic fraction and $\dfrac{m}{m^2+n^2}$ is $\dfrac{m+n}{m^2+n^2}$?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Let the required fraction be }F.\\
\Step{2}\;& F+\frac{m}{m^2+n^2}=\frac{m+n}{m^2+n^2}\\
\Step{3}\;& F=\frac{m+n-m}{m^2+n^2}=\frac{n}{m^2+n^2}.\\
\Step{4}\;& \boxed{F=\dfrac{n}{m^2+n^2}.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1(iv)}
\textcolor{gold}{\bfseries Question:} What is the product of an algebraic fraction and its reciprocal?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\boxed{1}
\qquad (\text{provided the fraction is non-zero}).
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{14x^2-7x}{12x^3+24x^2}\times \frac{x^2+2x}{2x-1}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 14x^2-7x=7x(2x-1),\qquad 12x^3+24x^2=12x^2(x+2),\\
& x^2+2x=x(x+2).\\
\Step{2}\;& \frac{7x(2x-1)}{12x^2(x+2)}\times \frac{x(x+2)}{2x-1}.\\
\Step{3}\;& \text{Cancel }(2x-1),\ (x+2),\ \text{and }x.\\
\Step{4}\;& \boxed{\frac{7}{12}}.
\end{aligned}
\]
\end{QAPair}

% Q3
\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{a^2b^2+3ab}{4a^2-1}\times \frac{2a+1}{ab+3}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& a^2b^2+3ab=ab(ab+3),\qquad 4a^2-1=(2a-1)(2a+1).\\
\Step{2}\;& \frac{ab(ab+3)}{(2a-1)(2a+1)}\times \frac{2a+1}{ab+3}.\\
\Step{3}\;& \text{Cancel }(ab+3)\text{ and }(2a+1).\\
\Step{4}\;& \boxed{\frac{ab}{2a-1}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{a-b}{a^2+ab}\times \frac{a^4-b^4}{a^2-2ab+b^2}\times \frac{a}{a^2+b^2}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& a^2+ab=a(a+b),\qquad a^2-2ab+b^2=(a-b)^2,\\
& a^4-b^4=(a-b)(a+b)(a^2+b^2).\\
\Step{2}\;& \frac{a-b}{a(a+b)}\times \frac{(a-b)(a+b)(a^2+b^2)}{(a-b)^2}\times \frac{a}{a^2+b^2}.\\
\Step{3}\;& \text{Cancel }(a-b),\ (a+b),\ a,\ (a^2+b^2).\\
\Step{4}\;& \boxed{1}.
\end{aligned}
\]
\end{QAPair}

% Q5
\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{6x^2y^2}{x^2-y^2}\div \frac{3xy}{x+y}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{6x^2y^2}{x^2-y^2}\div \frac{3xy}{x+y}
= \frac{6x^2y^2}{x^2-y^2}\times \frac{x+y}{3xy}.\\
\Step{2}\;& x^2-y^2=(x-y)(x+y).\\
\Step{3}\;& \frac{6x^2y^2}{(x-y)(x+y)}\times \frac{x+y}{3xy}
= \frac{6x^2y^2}{x-y}\times \frac{1}{3xy}.\\
\Step{4}\;& \boxed{\frac{2xy}{x-y}}.
\end{aligned}
\]
\end{QAPair}

% Q6
\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{8a^3-1}{4a^3+2a^2}\div \frac{6a^2-13a+5}{15a-25}\times \frac{2a^4+a^3}{15a^2}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 8a^3-1=(2a-1)(4a^2+2a+1),\quad 4a^3+2a^2=2a^2(2a+1),\\
& 6a^2-13a+5=(3a-5)(2a-1),\quad 15a-25=5(3a-5),\\
& \frac{2a^4+a^3}{15a^2}=\frac{a^3(2a+1)}{15a^2}=\frac{a(2a+1)}{15}.\\
\Step{2}\;& \frac{(2a-1)(4a^2+2a+1)}{2a^2(2a+1)}
\times \frac{5(3a-5)}{(3a-5)(2a-1)}
\times \frac{a(2a+1)}{15}.\\
\Step{3}\;& \text{Cancel }(2a-1),\ (3a-5),\ (2a+1)\text{ and one }a.\\
\Step{4}\;& \frac{4a^2+2a+1}{2a^2}\times \frac{5a}{15}
=\frac{4a^2+2a+1}{2a^2}\times \frac{a}{3}.\\
\Step{5}\;& \boxed{\frac{4a^2+2a+1}{6a}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{x^2-8x-9}{x^2-17x+72}\times \frac{x^2-25}{x^2-1}\div \frac{x^2+4x-5}{x^2-9x+8}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& x^2-8x-9=(x-9)(x+1),\quad x^2-17x+72=(x-9)(x-8),\\
& x^2-25=(x-5)(x+5),\quad x^2-1=(x-1)(x+1),\\
& x^2+4x-5=(x+5)(x-1),\quad x^2-9x+8=(x-1)(x-8).\\
\Step{2}\;& \frac{(x-9)(x+1)}{(x-9)(x-8)}\times
\frac{(x-5)(x+5)}{(x-1)(x+1)}\times
\frac{(x-1)(x-8)}{(x+5)(x-1)}.\\
\Step{3}\;& \text{Cancel }(x-9),(x+1),(x-8),(x+5),(x-1).\\
\Step{4}\;& \boxed{\frac{x-5}{x-1}}.
\end{aligned}
\]
\end{QAPair}

% Q8
\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{1}{2x-3y}-\frac{x+y}{4x^2-9y^2}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 4x^2-9y^2=(2x-3y)(2x+3y).\\
\Step{2}\;& \frac{1}{2x-3y}=\frac{2x+3y}{(2x-3y)(2x+3y)}.\\
\Step{3}\;& \frac{2x+3y}{(2x-3y)(2x+3y)}-\frac{x+y}{(2x-3y)(2x+3y)}\\
&=\frac{(2x+3y)-(x+y)}{(2x-3y)(2x+3y)}=\frac{x+2y}{(2x-3y)(2x+3y)}.\\
\Step{4}\;& \boxed{\frac{x+2y}{(2x-3y)(2x+3y)}}.
\end{aligned}
\]
\end{QAPair}

% Q9
\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{1}{x(x-y)}+\frac{1}{y(x+y)}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \text{Common denominator }=xy(x-y)(x+y).\\
\Step{2}\;& \frac{1}{x(x-y)}=\frac{y(x+y)}{xy(x-y)(x+y)},\quad
\frac{1}{y(x+y)}=\frac{x(x-y)}{xy(x-y)(x+y)}.\\
\Step{3}\;& \frac{y(x+y)+x(x-y)}{xy(x-y)(x+y)}
=\frac{(xy+y^2)+(x^2-xy)}{xy(x-y)(x+y)}\\
&=\frac{x^2+y^2}{xy(x-y)(x+y)}.\\
\Step{4}\;& \boxed{\frac{x^2+y^2}{xy(x-y)(x+y)}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{5x+5}{3(2x-1)}+\frac{6-2x}{2(1-2x)}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1-2x=-(2x-1)\;\Rightarrow\;
\frac{6-2x}{2(1-2x)}=\frac{6-2x}{-2(2x-1)}=\frac{2x-6}{2(2x-1)}=\frac{x-3}{2x-1}.\\
\Step{2}\;& \frac{5x+5}{3(2x-1)}+\frac{x-3}{2x-1}
= \frac{5(x+1)}{3(2x-1)}+\frac{3(x-3)}{3(2x-1)}.\\
\Step{3}\;& \frac{5(x+1)+3(x-3)}{3(2x-1)}
=\frac{5x+5+3x-9}{3(2x-1)}=\frac{8x-4}{3(2x-1)}\\
&=\frac{4(2x-1)}{3(2x-1)}.\\
\Step{4}\;& \boxed{\frac{4}{3}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 11}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{2a}{2a-3}-\frac{5}{6a+9}-\frac{4(3a+2)}{3(4a^2-9)}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 6a+9=3(2a+3),\qquad 4a^2-9=(2a-3)(2a+3).\\
\Step{2}\;& \text{Common denominator }=3(2a-3)(2a+3)=3(4a^2-9).\\
\Step{3}\;& \frac{2a}{2a-3}=\frac{6a(2a+3)}{3(2a-3)(2a+3)},\\
& \frac{5}{6a+9}=\frac{5(2a-3)}{3(2a+3)(2a-3)},\qquad
\frac{4(3a+2)}{3(4a^2-9)}=\frac{4(3a+2)}{3(2a-3)(2a+3)}.\\
\Step{4}\;& \frac{6a(2a+3)-5(2a-3)-4(3a+2)}{3(2a-3)(2a+3)}\\
&=\frac{(12a^2+18a)+(-10a+15)+(-12a-8)}{3(2a-3)(2a+3)}
=\frac{12a^2-4a+7}{3(2a-3)(2a+3)}.\\
\Step{5}\;& \boxed{\frac{12a^2-4a+7}{3(4a^2-9)}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 12}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{5}{5+x-18x^2}-\frac{2}{2+5x+2x^2}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5+x-18x^2=(5-9x)(2x+1),\qquad 2+5x+2x^2=(2x+1)(x+2).\\
\Step{2}\;& \frac{5}{(5-9x)(2x+1)}-\frac{2}{(2x+1)(x+2)}.\\
\Step{3}\;& \text{Common denominator }=(2x+1)(5-9x)(x+2).\\
\Step{4}\;& \frac{5(x+2)-2(5-9x)}{(2x+1)(5-9x)(x+2)}
=\frac{(5x+10)-(10-18x)}{(2x+1)(5-9x)(x+2)}\\
&=\frac{23x}{(2x+1)(5-9x)(x+2)}.\\
\Step{5}\;& \boxed{\frac{23x}{(2x+1)(5-9x)(x+2)}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 13}
\textcolor{gold}{\bfseries Question:}
$\displaystyle \frac{1-p^2}{1+q}\times \frac{1-q^2}{p+p^2}\times \left(1+\frac{p}{1-p}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1-p^2=(1-p)(1+p),\qquad 1-q^2=(1-q)(1+q),\\
& p+p^2=p(1+p),\qquad 1+\frac{p}{1-p}=\frac{(1-p)+p}{1-p}=\frac{1}{1-p}.\\
\Step{2}\;& \frac{(1-p)(1+p)}{1+q}\times \frac{(1-q)(1+q)}{p(1+p)}\times \frac{1}{1-p}.\\
\Step{3}\;& \text{Cancel }(1+p),\ (1-p),\ (1+q).\\
\Step{4}\;& \boxed{\frac{1-q}{p}}.
\end{aligned}
\]
\end{QAPair}
