\begin{QuickBox}
{\color{cyan}\bfseries Quick laws of logarithms (useful)}\par\medskip
\begin{itemize}
\item \textbf{Product:} $\log(AB)=\log A+\log B$.
\item \textbf{Quotient:} $\log\!\left(\dfrac{A}{B}\right)=\log A-\log B$.
\item \textbf{Power:} $\log(A^k)=k\,\log A$.
\item \textbf{Root:} $\log\!\left(\sqrt[n]{A}\right)=\dfrac1n\,\log A$.
\item \textbf{Change of base:} $\log_a b=\dfrac{\log b}{\log a}$ \; (any common base for the logs on top/bottom).
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1 (i)}
\textcolor{gold}{\bfseries Question:} $\log(9t)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log(9t)=\log 9+\log t.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ii)}
\textcolor{gold}{\bfseries Question:} $\log\!\left(\dfrac{59}{s}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\left(\frac{59}{s}\right)=\log 59-\log s.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iii)}
\textcolor{gold}{\bfseries Question:} $\log\!\left(\dfrac{5pq^{2}}{xy^{3}}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\left(\frac{5pq^{2}}{xy^{3}}\right)
=\log(5pq^2)-\log(xy^3)\\
\Step{2}\;&= (\log 5+\log p+\log q^2)-(\log x+\log y^3)\\
\Step{3}\;&= \log 5+\log p+2\log q-\log x-3\log y.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iv)}
\textcolor{gold}{\bfseries Question:} $\log\!\sqrt{\dfrac{53.3}{46.4}}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\sqrt{\frac{53.3}{46.4}}
=\log\!\left(\left(\frac{53.3}{46.4}\right)^{\frac12}\right)\\
\Step{2}\;&=\frac12\log\!\left(\frac{53.3}{46.4}\right)
=\frac12(\log 53.3-\log 46.4).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (v)}
\textcolor{gold}{\bfseries Question:} $\log\!\left(\dfrac{5^{2}t^{5}a^{\frac13}}{\sqrt[3]{4.4\,t\,b^{3}}}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\left(\frac{5^{2}t^{5}a^{\frac13}}{\sqrt[3]{4.4\,t\,b^{3}}}\right)
=\log(5^{2}t^{5}a^{\frac13})-\log\!\left(\sqrt[3]{4.4\,t\,b^{3}}\right)\\
\Step{2}\;&= (\log 5^2+\log t^5+\log a^{\frac13})-\frac13\log(4.4\,t\,b^3)\\
\Step{3}\;&= (2\log 5+5\log t+\tfrac13\log a)-\frac13(\log 4.4+\log t+3\log b)\\
\Step{4}\;&= 2\log 5+\frac{14}{3}\log t+\frac13\log a-\frac13\log 4.4-\log b.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (vi)}
\textcolor{gold}{\bfseries Question:} $\log\!\sqrt[5]{\dfrac{7^{2}t^{3}p}{d^{6}b^{2}}}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\sqrt[5]{\frac{7^{2}t^{3}p}{d^{6}b^{2}}}
=\frac15\log\!\left(\frac{7^{2}t^{3}p}{d^{6}b^{2}}\right)\\
\Step{2}\;&=\frac15\Big(\log(7^{2}t^{3}p)-\log(d^{6}b^{2})\Big)\\
\Step{3}\;&=\frac15\Big(2\log 7+3\log t+\log p-6\log d-2\log b\Big).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (vii)}
\textcolor{gold}{\bfseries Question:} $\log\!\left(\left(\dfrac{\sqrt[3]{5.512\,p\,m^{\frac12}}}{\sqrt[4]{5.91\,a^{2}b}}\right)^{t}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log\!\left(\left(\frac{\sqrt[3]{5.512\,p\,m^{\frac12}}}{\sqrt[4]{5.91\,a^{2}b}}\right)^{t}\right)
=t\,\log\!\left(\frac{\sqrt[3]{5.512\,p\,m^{\frac12}}}{\sqrt[4]{5.91\,a^{2}b}}\right)\\
\Step{2}\;&=t\Bigg[\log\!\left(\sqrt[3]{5.512\,p\,m^{\frac12}}\right)-\log\!\left(\sqrt[4]{5.91\,a^{2}b}\right)\Bigg]\\
\Step{3}\;&=t\Bigg[\frac13\log(5.512\,p\,m^{\frac12})-\frac14\log(5.91\,a^{2}b)\Bigg]\\
\Step{4}\;&=t\Bigg[\frac13\Big(\log 5.512+\log p+\tfrac12\log m\Big)
-\frac14\Big(\log 5.91+2\log a+\log b\Big)\Bigg].
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (i)}
\textcolor{gold}{\bfseries Question:} $3\log x-5\log y$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&3\log x=\log x^{3},\quad 5\log y=\log y^{5}\\
\Step{2}\;&3\log x-5\log y=\log x^{3}-\log y^{5}
=\log\!\left(\frac{x^{3}}{y^{5}}\right).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (ii)}
\textcolor{gold}{\bfseries Question:} $\dfrac12\log t+\dfrac13\log r-\dfrac15\log s$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&\frac12\log t=\log t^{\frac12},\ \frac13\log r=\log r^{\frac13},\ \frac15\log s=\log s^{\frac15}\\
\Step{2}\;&\Rightarrow\ \frac12\log t+\frac13\log r-\frac15\log s
=\log\!\left(\frac{t^{\frac12}r^{\frac13}}{s^{\frac15}}\right).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (iii)}
\textcolor{gold}{\bfseries Question:} $\dfrac17\Big[\log 57.7-3\log 9.24+4\log 36.6-2\log 23.3\Big]$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&\log 57.7+4\log 36.6-3\log 9.24-2\log 23.3\\
&=\log\!\left(\frac{57.7\cdot(36.6)^{4}}{(9.24)^{3}\cdot(23.3)^{2}}\right)\\
\Step{2}\;&\Rightarrow\ \frac17[\cdots]
=\frac17\log\!\left(\frac{57.7\cdot(36.6)^{4}}{(9.24)^{3}\cdot(23.3)^{2}}\right)
=\log\!\left[\left(\frac{57.7\cdot(36.6)^{4}}{(9.24)^{3}\cdot(23.3)^{2}}\right)^{\frac17}\right].
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (iv)}
\textcolor{gold}{\bfseries Question:} $5\log 6-7\log 9.42+\dfrac13\log t-\dfrac12\log 32.2+\dfrac23\log a$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&5\log 6=\log 6^5,\quad -7\log 9.42=-\log(9.42)^7,\\
&\frac13\log t=\log t^{\frac13},\quad -\frac12\log 32.2=-\log(32.2)^{\frac12},\quad
\frac23\log a=\log a^{\frac23}\\
\Step{2}\;&\Rightarrow\ \text{given expression}
=\log\!\left(\frac{6^{5}\,t^{\frac13}\,a^{\frac23}}{(9.42)^{7}\,(32.2)^{\frac12}}\right).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2 (v)}
\textcolor{gold}{\bfseries Question:} $\dfrac54\log 37.74-\dfrac14\log 53.71+\dfrac14\log 28.83$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&\frac54\log 37.74=\frac14(5\log 37.74),\quad
-\frac14\log 53.71,\quad
+\frac14\log 28.83\\
\Step{2}\;&=\frac14\Big(5\log 37.74-\log 53.71+\log 28.83\Big)\\
\Step{3}\;&=\frac14\log\!\left(\frac{(37.74)^{5}\cdot 28.83}{53.71}\right)
=\log\!\left[\left(\frac{(37.74)^{5}\cdot 28.83}{53.71}\right)^{\frac14}\right].
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (i)}
\textcolor{gold}{\bfseries Question:} $\log_{2} 15$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_{2}15=\frac{\log 15}{\log 2}.\\
\Step{2}\;& \log 15=\log(3\cdot 5)=\log 3+\log 5.\\
\Step{3}\;& \text{Using common log values }(\log 2\approx0.3010,\ \log 3\approx0.4771,\ \log 5\approx0.6990):\\
&\log 15\approx 0.4771+0.6990=1.1761.\\
\Step{4}\;& \Rightarrow\ \log_{2}15\approx \frac{1.1761}{0.3010}\approx 3.9069.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (ii)}
\textcolor{gold}{\bfseries Question:} $\log_{9}\sqrt[3]{9}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt[3]{9}=9^{\frac13}.\\
\Step{2}\;& \log_{9}(9^{\frac13})=\frac13.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iii)}
\textcolor{gold}{\bfseries Question:} $\log_{3} 65$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_{3}65=\frac{\log 65}{\log 3}.\\
\Step{2}\;& \text{From log tables (characteristic + mantissa): } \log 65 \approx 1.8129,\ \log 3\approx 0.4771.\\
\Step{3}\;& \Rightarrow\ \log_{3}65\approx \frac{1.8129}{0.4771}\approx 3.7997.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iv)}
\textcolor{gold}{\bfseries Question:} $\log_{\sqrt{3}} 72.34$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_{\sqrt{3}}72.34=\frac{\log 72.34}{\log(\sqrt{3})}.\\
\Step{2}\;& \log(\sqrt{3})=\log(3^{\frac12})=\frac12\log 3.\\
\Step{3}\;& \text{Using log tables: } \log 72.34\approx 1.8594,\ \log 3\approx 0.4771
\ \Rightarrow\ \log(\sqrt{3})\approx 0.2386.\\
\Step{4}\;& \Rightarrow\ \log_{\sqrt{3}}72.34\approx \frac{1.8594}{0.2386}\approx 7.7942.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (v)}
\textcolor{gold}{\bfseries Question:} $\log_{\sqrt{7}} 343$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 343=7^{3},\quad \sqrt{7}=7^{\frac12}.\\
\Step{2}\;& \log_{7^{\frac12}}(7^{3})=\frac{3}{\frac12}=6.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4 (i)}
\textcolor{gold}{\bfseries Question:} If $\log_b 2=0.3010,\ \log_b 3=0.4771,\ \log_b 5=0.6990$, find $\log_b\!\left(\dfrac65\right)$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_b\!\left(\frac65\right)=\log_b 6-\log_b 5.\\
\Step{2}\;& \log_b 6=\log_b(2\cdot 3)=\log_b2+\log_b3=0.3010+0.4771=0.7781.\\
\Step{3}\;& \Rightarrow\ \log_b\!\left(\frac65\right)=0.7781-0.6990=0.0791.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4 (ii)}
\textcolor{gold}{\bfseries Question:} $\log_b\!\left(\dfrac{100}{9}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_b\!\left(\frac{100}{9}\right)=\log_b 100-\log_b 9.\\
\Step{2}\;& 100=2^2\cdot 5^2\Rightarrow \log_b100=2\log_b2+2\log_b5
=2(0.3010)+2(0.6990)=2.0000.\\
\Step{3}\;& 9=3^2\Rightarrow \log_b 9=2\log_b3=2(0.4771)=0.9542.\\
\Step{4}\;& \Rightarrow\ \log_b\!\left(\frac{100}{9}\right)=2.0000-0.9542=1.0458.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4 (iii)}
\textcolor{gold}{\bfseries Question:} $\log_b\!\left(\dfrac{\sqrt[3]{450}}{\sqrt{27}}\right)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_b\!\left(\frac{\sqrt[3]{450}}{\sqrt{27}}\right)
=\log_b(450^{\frac13})-\log_b(27^{\frac12})
=\frac13\log_b450-\frac12\log_b27.\\
\Step{2}\;& 450=2\cdot 3^2\cdot 5^2 \Rightarrow \log_b450=\log_b2+2\log_b3+2\log_b5\\
&=0.3010+2(0.4771)+2(0.6990)=0.3010+0.9542+1.3980=2.6532.\\
\Step{3}\;& 27=3^3 \Rightarrow \log_b27=3\log_b3=3(0.4771)=1.4313.\\
\Step{4}\;& \Rightarrow\ \frac13\log_b450-\frac12\log_b27
=\frac{2.6532}{3}-\frac{1.4313}{2}=0.8844-0.7157\approx 0.1688.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4 (iv)}
\textcolor{gold}{\bfseries Question:} $\log_b(0.024)$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 0.024=\frac{24}{1000}=\frac{2^3\cdot 3}{2^3\cdot 5^3}=\frac{3}{5^3}.\\
\Step{2}\;& \log_b(0.024)=\log_b3-3\log_b5
=0.4771-3(0.6990)=0.4771-2.0970=-1.6199.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4 (v)}
\textcolor{gold}{\bfseries Question:} $\log_b\!\sqrt[7]{\dfrac{5^{2}}{5}}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \sqrt[7]{\frac{5^2}{5}}=\sqrt[7]{5}=5^{\frac17}.\\
\Step{2}\;& \log_b(5^{\frac17})=\frac17\log_b5=\frac{0.6990}{7}\approx 0.0999.
\end{aligned}
\]
\end{QAPair}
