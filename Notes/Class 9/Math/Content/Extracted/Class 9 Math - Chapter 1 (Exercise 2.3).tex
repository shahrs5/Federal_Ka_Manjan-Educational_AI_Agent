\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (Characteristic \& Mantissa)}\par\medskip
\begin{itemize}
\item We use \textbf{common logarithm}: $\log(x)=\log_{10}(x)$.
\item Write the number in \textbf{standard/scientific form}: $N=a\times 10^{k}$ where $1\le a<10$.
\item Then
\[
\log(N)=\log(a\times 10^{k})=\log(a)+k.
\]
\item \textbf{Characteristic} $C$ is the \textbf{integer part} (coming from $k$), and \textbf{Mantissa} $M$ is the \textbf{decimal part}:
\[
\log(N)=C+M,\qquad 0\le M<1.
\]
\item For $0<N<1$, the characteristic is negative. Using \textbf{bar notation}:
\[
\log(N)=\barc{k}+M \quad \text{means} \quad \barc{k}+M = -k + M.
\]
(So the mantissa stays positive.)
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} Find $\log_{10}(5313)$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5313 = 5.313\times 10^{3}.\\
\Step{2}\;& \text{Characteristic } C = 3.\\
\Step{3}\;& \text{Mantissa } M = \log_{10}(5.313)\approx 0.72534.\\
\Step{4}\;& \log_{10}(5313)=C+M=3+0.72534=3.72534.
\end{aligned}
\]
\end{QAPair}

\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \log_{10}(x)\ \text{(in real numbers) is defined only for } x>0.\\
\Step{2}\;& \text{Since } -4<0,\ \log_{10}(-4)\ \textbf{is not possible in real numbers.}
\end{aligned}
\]
