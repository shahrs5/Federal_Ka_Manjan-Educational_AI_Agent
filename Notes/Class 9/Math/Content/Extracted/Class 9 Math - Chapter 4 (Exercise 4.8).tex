\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Factorization:} $x^2+bx+c=(x+m)(x+n)$ where $m+n=b$ and $mn=c$.
\item \textbf{Rectangle:} Area $A=lb$, \;Perimeter $P=2(l+b)$.
\item \textbf{Square:} Area $A=s^2$, \;Boundary (perimeter) $=4s$.
\item \textbf{Cube:} Volume $V=s^3$, \;Surface area $=6s^2$.
\item \textbf{Circle:} Area $A=\pi r^2$.
\item \textbf{Difference of squares:} $a^2-b^2=(a-b)(a+b)$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} Area of a rectangular veranda is $(x^2-2x-3)\,m^2$. Find:\\
(a) expressions for both dimensions, (b) perimeter, (c) cost of fencing @ Rs.\ 200/m, (d) cost of carpeting @ Rs.\ 250/$m^2$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& x^2-2x-3=(x-3)(x+1).\\
\Step{2}\;& \text{So possible dimensions are } (x-3)\,m \text{ and } (x+1)\,m.\\
\Step{3}\;& P=2\bigl[(x-3)+(x+1)\bigr]\,m=2(2x-2)\,m=(4x-4)\,m.\\
\Step{4}\;& C_f=200\cdot P=200(4x-4)=800x-800\ \text{Rs}.\\
\Step{5}\;& C_c=250\cdot \text{Area}=250(x^2-2x-3)\ \text{Rs}.
\end{aligned}
\]
\[
\boxed{
\begin{aligned}
\text{Dimensions: }& (x-3)m,\ (x+1)m\\
\text{Perimeter: }& P=(4x-4)m\\
\text{Fencing cost: }& C_f=800x-800\ \text{Rs}\\
\text{Carpeting cost: }& C_c=250(x^2-2x-3)\ \text{Rs}
\end{aligned}}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} Area of a square surface is $(25x^2-30x+9)\,m^2$. Find:\\
(a) length of the surface, (b) boundary, (c) cost of polishing @ Rs.\ 75/$m^2$, (d) cost of edging around 2 sides @ Rs.\ 28/m.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&25x^2-30x+9=(5x-3)^2.\\
\Step{2}\;& \text{Side length } s=\sqrt{(5x-3)^2}=(5x-3)\,m.\\
\Step{3}\;& \text{Boundary}=4s=4(5x-3)=(20x-12)\,m.\\
\Step{4}\;& C_p=75\cdot \text{Area}=75(25x^2-30x+9)\ \text{Rs}.\\
\Step{5}\;& \text{2 sides length}=2s=2(5x-3)=(10x-6)\,m,\\
& C_e=28(10x-6)=280x-168\ \text{Rs}.
\end{aligned}
\]
\[
\boxed{
\begin{aligned}
\text{Side length: }& s=(5x-3)m\\
\text{Boundary: }& (20x-12)m\\
\text{Polishing cost: }& C_p=75(25x^2-30x+9)\ \text{Rs}\\
\text{Edging cost (2 sides): }& C_e=280x-168\ \text{Rs}
\end{aligned}}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} Volume of a cubical oil tank is $(125x^3-150x^2+60x-8)\,m^3$. Find:\\
(a) height (side) of tank, (b) surface area, (c) cost of painting outside @ Rs.\ 32/$m^2$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&125x^3-150x^2+60x-8=(5x-2)^3.\\
\Step{2}\;&\text{For a cube, }V=s^3\Rightarrow s=\sqrt[3]{V}=(5x-2)\,m.\\
\Step{3}\;&\text{Surface area}=6s^2=6(5x-2)^2\ m^2.\\
\Step{4}\;&\text{Painting cost}=32\cdot \text{surface area}=32\cdot 6(5x-2)^2=192(5x-2)^2\ \text{Rs}.\\[4pt]
&\boxed{s=(5x-2)m;\ \text{S.A.}=6(5x-2)^2;\ \text{Cost}=192(5x-2)^2.}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} Areas of two wheels are
\[
A_1=\pi x^2-6\pi x+9\pi,\qquad
A_2=\pi x^2-10\pi x+25\pi.
\]
Find radii of both wheels.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&A_1=\pi(x^2-6x+9)=\pi(x-3)^2.\\
\Step{2}\;&A_2=\pi(x^2-10x+25)=\pi(x-5)^2.\\
\Step{3}\;&\text{Since }A=\pi r^2,\ \ r_1=x-3,\ \ r_2=x-5\ (\text{taking positive radii}).\\[4pt]
&\boxed{r_1=x-3,\qquad r_2=x-5\ \ (\text{usually }x>5\text{ so both are positive}).}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} Two squared shape pressers have areas $25m^2$ and $36n^2$. Their difference gives a rectangular presser. Find dimensions of the rectangular presser.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&25m^2=(5m)^2,\qquad 36n^2=(6n)^2.\\
\Step{2}\;&\text{Difference of areas }=36n^2-25m^2=(6n)^2-(5m)^2.\\
\Step{3}\;&(6n)^2-(5m)^2=(6n-5m)(6n+5m).\\
\Step{4}\;&\text{So the rectangle can have dimensions }(6n-5m)\ \text{and}\ (6n+5m).\\[4pt]
&\boxed{\text{Dimensions: }(6n-5m)\ \text{ and }\ (6n+5m).}
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} Distance covered by a missile is $(x^2+5x+6)\,m$. Find possible expressions for\\
(a) speed of missile, (b) time to reach the target.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;&x^2+5x+6=(x+2)(x+3).\\
\Step{2}\;&\text{Distance}=\text{Speed}\times\text{Time}.\\
\Step{3}\;&\text{So one possible pair is: Speed }=(x+2)\ \text{and Time }=(x+3).\\
\Step{4}\;&\text{Another possible pair is: Speed }=(x+3)\ \text{and Time }=(x+2).\\[4pt]
&\boxed{\text{Possible: }(v,t)=((x+2),(x+3))\ \text{or}\ ((x+3),(x+2)).}
\end{aligned}
\]
\end{QAPair}
