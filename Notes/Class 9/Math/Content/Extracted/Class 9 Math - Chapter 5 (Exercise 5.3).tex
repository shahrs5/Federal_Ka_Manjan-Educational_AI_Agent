\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Definition:} $|A|=\begin{cases}A,&A\ge 0\\-A,&A<0\end{cases}$.
\item \textbf{Key rule (two cases):} If $|A|=b$ with $b\ge 0$, then $A=b$ \textbf{or} $A=-b$.
\item \textbf{No-solution rule:} If $|A|=b$ with $b<0$, then \textbf{no real solution}.
\item \textbf{Isolate first:} If $c|A|+d=e$, first make $|A|=\dfrac{e-d}{c}$, then apply the two-case rule.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} $|x|=\dfrac{5}{3}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |x|=\frac{5}{3}\ \Rightarrow\ x=\frac{5}{3}\ \text{or}\ x=-\frac{5}{3}.\\
\Step{2}\;& \boxed{\,x\in\left\{-\frac{5}{3},\ \frac{5}{3}\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} $|x+2|=6$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& x+2=6 \ \text{or}\ x+2=-6.\\
\Step{2}\;& x=4 \ \text{or}\ x=-8.\\
\Step{3}\;& \boxed{\,x\in\{-8,\ 4\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} $|5y-1|=9$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5y-1=9 \ \text{or}\ 5y-1=-9.\\
\Step{2}\;& 5y=10 \Rightarrow y=2,\qquad 5y=-8 \Rightarrow y=-\frac{8}{5}.\\
\Step{3}\;& \boxed{\,y\in\left\{2,\ -\frac{8}{5}\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}

\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |6-3y|=0 \Rightarrow 6-3y=0 \qquad (\text{only possible when inside }=0).\\

\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} $3|z-2|-4=-2$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 3|z-2|-4=-2 \Rightarrow 3|z-2|=2.\\
\Step{2}\;& |z-2|=\frac{2}{3}.\\
\Step{3}\;& z-2=\frac{2}{3}\ \text{or}\ z-2=-\frac{2}{3}.\\
\Step{4}\;& z=\frac{8}{3}\ \text{or}\ z=\frac{4}{3}.\\
\Step{5}\;& \boxed{\,z\in\left\{\frac{4}{3},\ \frac{8}{3}\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:} $\dfrac{|4x|}{3}=12$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{|4x|}{3}=12 \Rightarrow |4x|=36.\\
\Step{2}\;& |4x|=4|x| \Rightarrow 4|x|=36 \Rightarrow |x|=9.\\
\Step{3}\;& x=9 \ \text{or}\ x=-9.\\
\Step{4}\;& \boxed{\,x\in\{-9,\ 9\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} $|5x|+10=5$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |5x|+10=5 \Rightarrow |5x|=-5.\\
\Step{2}\;& |5x|\ge 0 \text{ for all real }x,\ \text{so it cannot equal }-5.\\
\Step{3}\;& \boxed{\,\text{No real solution (solution set }=\varnothing)\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 11}
\textcolor{gold}{\bfseries Question:} $\dfrac{|1-2y|}{4}=3$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{|1-2y|}{4}=3 \Rightarrow |1-2y|=12.\\
\Step{2}\;& 1-2y=12 \ \text{or}\ 1-2y=-12.\\
\Step{3}\;& -2y=11 \Rightarrow y=-\frac{11}{2},\qquad -2y=-13 \Rightarrow y=\frac{13}{2}.\\
\Step{4}\;& \boxed{\,y\in\left\{-\frac{11}{2},\ \frac{13}{2}\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 12}
\textcolor{gold}{\bfseries Question:} $\dfrac{|x+1|}{2}=\dfrac{|2x-1|}{3}$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& \frac{|x+1|}{2}=\frac{|2x-1|}{3}\ \Rightarrow\ 3|x+1|=2|2x-1|.\\
\Step{2}\;& \text{Break points: }x=-1\ (\text{from }x+1),\quad x=\frac12\ (\text{from }2x-1).\\
\Step{3}\;& \textbf{Case A: }x\ge \frac12:\ 3(x+1)=2(2x-1)\\
&\Rightarrow 3x+3=4x-2 \Rightarrow x=5.\\
\Step{4}\;& \textbf{Case B: }-1\le x<\frac12:\ 3(x+1)=2\bigl(-(2x-1)\bigr)=2(-2x+1)\\
&\Rightarrow 3x+3=-4x+2 \Rightarrow 7x=-1 \Rightarrow x=-\frac17.\\
\Step{5}\;& \textbf{Case C: }x<-1:\ 3\bigl(-(x+1)\bigr)=2\bigl(-(2x-1)\bigr)\Rightarrow x=5\ (\text{reject since }x<-1).\\
\Step{6}\;& \boxed{\,x\in\left\{5,\ -\frac17\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 13}
\textcolor{gold}{\bfseries Question:} $|5x-3|=|x+7|$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& |A|=|B|\ \Rightarrow\ A=B\ \text{or}\ A=-B.\\
\Step{2}\;& 5x-3=x+7 \Rightarrow 4x=10 \Rightarrow x=\frac52.\\
\Step{3}\;& 5x-3=-(x+7) \Rightarrow 5x-3=-x-7 \Rightarrow 6x=-4 \Rightarrow x=-\frac23.\\
\Step{4}\;& \boxed{\,x\in\left\{-\frac23,\ \frac52\right\}\,}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 14}
\textcolor{gold}{\bfseries Question:} $|z+3|-3=5-|z+3|$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
Let $t=|z+3|$ (so $t\ge 0$).
\[
\begin{aligned}
\Step{1}\;& t-3=5-t.\\
\Step{2}\;& 2t=8 \Rightarrow t=4.\\
\Step{3}\;& |z+3|=4 \Rightarrow z+3=4 \ \text{or}\ z+3=-4.\\
\Step{4}\;& z=1 \ \text{or}\ z=-7.\\
\Step{5}\;& \boxed{\,z\in\{-7,\ 1\}\,}.
\end{aligned}
\]
\end{QAPair}
