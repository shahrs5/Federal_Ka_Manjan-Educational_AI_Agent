\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (useful)}\par\medskip
\begin{itemize}
\item \textbf{Angle units:} $1\Deg = 60\Min$, \quad $1\Min = 60\Sec$, \quad so $1\Deg = 3600\Sec$.
\item \textbf{To seconds:} $D\Deg\,M\Min\,S\Sec \;\Rightarrow\; (3600D+60M+S)\Sec$.
\item \textbf{To minutes:} $D\Deg\,M\Min\,S\Sec \;\Rightarrow\; (60D+M+\tfrac{S}{60})\Min$.
\item \textbf{To degrees:} $D\Deg\,M\Min\,S\Sec \;\Rightarrow\; \left(D+\tfrac{M}{60}+\tfrac{S}{3600}\right)\Deg$.
\item \textbf{Degrees $\leftrightarrow$ radians:} $\theta(\text{rad})=\theta(\Deg)\cdot \dfrac{\pi}{180}$,\quad
$\theta(\Deg)=\theta(\text{rad})\cdot \dfrac{180}{\pi}$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1 (i)}
\textcolor{gold}{\bfseries Question:} Convert $5\Deg$ into seconds.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1\Deg = 3600\Sec.\\
\Step{2}\;& 5\Deg = 5\times 3600\Sec = 18000\Sec.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (ii)}
\textcolor{gold}{\bfseries Question:} Convert $30\Min$ into seconds.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1\Min = 60\Sec.\\
\Step{2}\;& 30\Min = 30\times 60\Sec = 1800\Sec.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iii)}
\textcolor{gold}{\bfseries Question:} Convert $10\Deg 30\Min$ into seconds.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 10\Deg = 10\times 3600\Sec = 36000\Sec.\\
\Step{2}\;& 30\Min = 30\times 60\Sec = 1800\Sec.\\
\Step{3}\;& \text{Total} = 36000\Sec + 1800\Sec = 37800\Sec.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 1 (iv)}
\textcolor{gold}{\bfseries Question:} Convert $20\Deg 20\Min 20\Sec$ into seconds.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 20\Deg = 20\times 3600\Sec = 72000\Sec.\\
\Step{2}\;& 20\Min = 20\times 60\Sec = 1200\Sec.\\
\Step{3}\;& \text{Total} = 72000\Sec + 1200\Sec + 20\Sec = 73220\Sec.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (i)}
\textcolor{gold}{\bfseries Question:} Convert $135\Min$ into degrees (correct to 4 decimals).\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1\Deg = 60\Min \;\Rightarrow\; 135\Min = \frac{135}{60}\Deg.\\
\Step{2}\;& \frac{135}{60}=2.25 \;\Rightarrow\; 135\Min = 2.2500\Deg.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (ii)}
\textcolor{gold}{\bfseries Question:} Convert $150\Sec$ into degrees (correct to 4 decimals).\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1\Deg = 3600\Sec \;\Rightarrow\; 150\Sec = \frac{150}{3600}\Deg.\\
\Step{2}\;& \frac{150}{3600}=0.041666\ldots \;\Rightarrow\; 150\Sec \approx 0.0417\Deg.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iii)}
\textcolor{gold}{\bfseries Question:} Convert $60\Deg 60\Min$ into degrees (correct to 4 decimals).\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 60\Min = 1\Deg.\\
\Step{2}\;& 60\Deg 60\Min = 60\Deg + 1\Deg = 61.0000\Deg.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3 (iv)}
\textcolor{gold}{\bfseries Question:} Convert $45\Deg 45\Min 45\Sec$ into degrees (correct to 4 decimals).\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 45\Deg 45\Min 45\Sec
= \left(45+\frac{45}{60}+\frac{45}{3600}\right)\Deg.\\
\Step{2}\;& = \left(45+0.75+0.0125\right)\Deg = 45.7625\Deg.\\
\Step{3}\;& \Rightarrow\; 45\Deg 45\Min 45\Sec = 45.7625\Deg.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (i)}

\tcblower
\textcolor{green}{\bfseries Answer:}

\begin{aligned}

\Step{2}\;& 45\Deg \Rightarrow 45\cdot \frac{\pi}{180}=\frac{\pi}{4}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (ii)}

\[
\begin{aligned}
\Step{1}\;& 150\Deg \Rightarrow 150\cdot \frac{\pi}{180}=\frac{150}{180}\pi.\\
\Step{2}\;& =\frac{5\pi}{6}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5 (iii)}

\Step{2}\;& \Rightarrow \theta = \frac{121}{2}\cdot \frac{\pi}{180}

\end{aligned}
\]

\begin{QAPair}{Question 6 (i)}
\textcolor{gold}{\bfseries Question:} Convert $270\Deg$ to radians. Use $\pi=3.1416$.\\

\textcolor{green}{\bfseries Answer:}
\[

\Step{2}\;& \pi=3.1416 \Rightarrow \theta = \frac{3}{2}(3.1416)=4.7124\ \text{rad}.
\end{aligned}

\begin{QAPair}{Question 7 (v)}

\[
\begin{aligned}

\Step{2}\;& \text{Minutes: } 0.2956455\times 60 \approx 17.7387319\Min.\\
\Step{3}\;& \text{Seconds: } 0.7387319\times 60 \approx 44.3239114\Sec.\\
\Step{4}\;& \Rightarrow\; 1\ \text{rad} \approx 57\Deg\,17\Min\,44.32\Sec.
\end{aligned}
\]
\end{QAPair}
