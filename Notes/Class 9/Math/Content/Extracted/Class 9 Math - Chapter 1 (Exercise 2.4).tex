\begin{QuickBox}
{\color{cyan}\bfseries Quick formulas (Characteristic \& Mantissa)}\par\medskip
\begin{itemize}
\item \textbf{Common antilog:} \(\text{Antilog}(x)=10^{x}\) (base \(10\)).
\item \textbf{Split:} If \(x=C+M\) where \(C\in\mathbb{Z}\) is the \textbf{characteristic} and \(0\le M<1\) is the \textbf{mantissa}, then
\[
10^{C+M}=10^{C}\cdot 10^{M}.
\]
\item \textbf{Use table on mantissa:} Find \(10^{M}\) from an \textbf{antilog table}, then multiply by \(10^C\) (shift decimal).
\item \textbf{Bar notation (negative characteristic):}
\[
\barc{n}.M \;=\; -n + M
\quad\Rightarrow\quad
10^{\barc{n}.M}=10^{-n}\cdot 10^{M}
\ \ (\text{shift decimal } n \text{ places left}).
\]
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1}
\textcolor{gold}{\bfseries Question:} Find \(\text{Antilog}(2.4324)\).\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 2.4324 = C+M = 2 + 0.4324 \quad (C=2,\ M=0.4324).\\
\Step{2}\;& 10^{2.4324}=10^{2}\cdot 10^{0.4324}.\\
\Step{3}\;& \text{From antilog table: } 10^{0.4324}\approx 2.7064.\\
\Step{4}\;& 10^{2.4324}\approx 10^{2}\cdot 2.7064 = 270.64 \approx 270.6.
\end{aligned}
\]
\end{QAPair}
