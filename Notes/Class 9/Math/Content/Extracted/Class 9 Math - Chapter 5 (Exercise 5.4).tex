\begin{QuickBox}
{\color{cyan}\bfseries Quick rules for inequalities}\par\medskip
\begin{itemize}
\item \textbf{Add/Subtract same number:} inequality sign does \textbf{not} change.
\item \textbf{Multiply/Divide by positive:} inequality sign does \textbf{not} change.
\item \textbf{Multiply/Divide by negative:} \textbf{flip the sign} ($<$ becomes $>$, $\le$ becomes $\ge$, etc.).
\item \textbf{Compound:} ``and'' means \textbf{intersection}; ``or'' means \textbf{union}.
\item \textbf{Set notation:} $\{x\in\R:\text{condition}\}$.
\end{itemize}
\end{QuickBox}

\begin{QAPair}{Question 1(a)}
\textcolor{gold}{\bfseries Question:} Check whether the given value satisfies the inequality.\\
(i) $5y-12>0;\ y=3$ \quad
(ii) $4-2x\le 9;\ x=-3$ \quad
(iii) $5-2x>-4x+5;\ x=4$\\
(iv) $3(z+4)\le 6;\ z=-2$ \quad
(v) $5(x-2)\ge 9x-3(2x-4);\ x=11$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\begin{enumerate}[label=\textbf{(\roman*)}]
\item \Step{1} Put $y=3$: $5(3)-12=15-12=3$.\\
\Step{2} $3>0$ is \textbf{true}. \ $\Rightarrow$ \boxed{\text{Satisfies}}.

\item \Step{1} Put $x=-3$: $4-2(-3)=4+6=10$.\\
\Step{2} $10\le 9$ is \textbf{false}. \ $\Rightarrow$ \boxed{\text{Does not satisfy}}.

\item \Step{1} Put $x=4$: LHS $=5-2(4)=5-8=-3$, RHS $=-4(4)+5=-16+5=-11$.\\
\Step{2} $-3>-11$ is \textbf{true}. \ $\Rightarrow$ \boxed{\text{Satisfies}}.

\item \Step{1} Put $z=-2$: $3(z+4)=3(-2+4)=3(2)=6$.\\
\Step{2} $6\le 6$ is \textbf{true}. \ $\Rightarrow$ \boxed{\text{Satisfies}}.

\item \Step{1} Put $x=11$: LHS $=5(11-2)=5(9)=45$.\\
RHS $=9(11)-3(2\cdot 11-4)=99-3(18)=99-54=45$.\\
\Step{2} $45\ge 45$ is \textbf{true}. \ $\Rightarrow$ \boxed{\text{Satisfies}}.
\end{enumerate}
\end{QAPair}

\begin{QAPair}{Question 1(b)}
\textcolor{gold}{\bfseries Question:} Write each solution in set notation form.\\
(i) $2\le x\le 5,\ x\in\N$ \quad
(ii) $y<7,\ y\in\N$ \quad
(iii) $z\le -3,\ z\in\R$\\
(iv) $x\le 4,\ x\in\W$ \quad
(v) $-4<x<-\dfrac{3}{2},\ x\in\R$\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\text{(i)}\;& \{x\in\N:2\le x\le 5\}=\{2,3,4,5\}.\\
\text{(ii)}\;& \{y\in\N:y<7\}=\{1,2,3,4,5,6\}.\\
\text{(iii)}\;& \{z\in\R:z\le -3\}.\\
\text{(iv)}\;& \{x\in\W:x\le 4\}=\{0,1,2,3,4\}.\\
\text{(v)}\;& \{x\in\R:-4<x<-\tfrac{3}{2}\}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 2}
\textcolor{gold}{\bfseries Question:} Solve $3x-2<7,\ x\in\N$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 3x-2<7 \;\Rightarrow\; 3x<9.\\
\Step{2}\;& x<3.\\
\Step{3}\;& x\in\N \Rightarrow x=\boxed{1,2}\ \ (\text{i.e., }\{1,2\}).
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 3}
\textcolor{gold}{\bfseries Question:} Solve $6x-5\le 35-2x,\ x\in\W$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 6x-5\le 35-2x \;\Rightarrow\; 8x\le 40.\\
\Step{2}\;& x\le 5.\\
\Step{3}\;& x\in\W \Rightarrow \boxed{\{0,1,2,3,4,5\}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 4}
\textcolor{gold}{\bfseries Question:} Solve $16-5y<4(y-1)-7,\ y\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 4(y-1)-7=4y-4-7=4y-11.\\
\Step{2}\;& 16-5y<4y-11.\\
\Step{3}\;& 16+11<4y+5y \;\Rightarrow\; 27<9y.\\
\Step{4}\;& y>3 \;\Rightarrow\; \boxed{\{y\in\R:y>3\}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 5}
\textcolor{gold}{\bfseries Question:} Solve $10-(7-y)\ge 3y-9,\ y\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 10-(7-y)=10-7+y=y+3.\\
\Step{2}\;& y+3\ge 3y-9.\\
\Step{3}\;& 12\ge 2y \;\Rightarrow\; y\le 6.\\
\Step{4}\;& \boxed{\{y\in\R:y\le 6\}}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 6}
\textcolor{gold}{\bfseries Question:} Solve $5-3x<11$ \textbf{or} $2x+3<-9,\ x\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5-3x<11 \Rightarrow -3x<6 \Rightarrow x>-2 \quad (\text{flip sign}).\\
\Step{2}\;& 2x+3<-9 \Rightarrow 2x<-12 \Rightarrow x<-6.\\
\Step{3}\;& \text{OR } \Rightarrow \boxed{(-\infty,-6)\ \cup\ (-2,\infty)}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 7}
\textcolor{gold}{\bfseries Question:} Solve $2x+3\le 9$ \textbf{and} $x-5>-6,\ x\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 2x+3\le 9 \Rightarrow 2x\le 6 \Rightarrow x\le 3.\\
\Step{2}\;& x-5>-6 \Rightarrow x>-1.\\
\Step{3}\;& \text{AND } \Rightarrow \boxed{(-1,3] }.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 8}
\textcolor{gold}{\bfseries Question:} Solve $1\le 7-3x\le 22,\ x\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1\le 7-3x\le 22.\\
\Step{2}\;& \text{Subtract }7:\quad -6\le -3x\le 15.\\
\Step{3}\;& \text{Divide by }-3\ (\text{flip both signs}):\quad 2\ge x\ge -5.\\
\Step{4}\;& \boxed{-5\le x\le 2}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 9}
\textcolor{gold}{\bfseries Question:} Solve $3x+21<1-x$ \textbf{or} $3x+8>3-2x,\ x\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 3x+21<1-x \Rightarrow 4x<-20 \Rightarrow x<-5.\\
\Step{2}\;& 3x+8>3-2x \Rightarrow 5x>-5 \Rightarrow x>-1.\\
\Step{3}\;& \text{OR } \Rightarrow \boxed{(-\infty,-5)\ \cup\ (-1,\infty)}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 10}
\textcolor{gold}{\bfseries Question:} Solve $1-5x>16$ \textbf{and} $3-\dfrac{3x}{2}\le 9,\ x\in\R$.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 1-5x>16 \Rightarrow -5x>15 \Rightarrow x<-3 \quad (\text{flip sign}).\\
\Step{2}\;& 3-\frac{3x}{2}\le 9 \Rightarrow -\frac{3x}{2}\le 6
\Rightarrow -3x\le 12 \Rightarrow x\ge -4.\\
\Step{3}\;& \text{AND } \Rightarrow \boxed{[-4,-3)}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 11}
\textcolor{gold}{\bfseries Question:} The sum of five times a number $x$ and $10$ is less than $-35$ or greater than $-5$.
What real numbers does $x$ represent?\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 5x+10<-35 \ \text{or}\ 5x+10>-5.\\
\Step{2}\;& 5x<-45 \Rightarrow x<-9.\\
\Step{3}\;& 5x>-15 \Rightarrow x>-3.\\
\Step{4}\;& \boxed{(-\infty,-9)\ \cup\ (-3,\infty)}.
\end{aligned}
\]
\end{QAPair}

\begin{QAPair}{Question 12}
\textcolor{gold}{\bfseries Question:} Two times a number decreased by $5$ is greater than or equal to the number increased by $8$.
Find the possible values for the number.\\
\tcblower
\textcolor{green}{\bfseries Answer:}
\[
\begin{aligned}
\Step{1}\;& 2x-5\ge x+8.\\
\Step{2}\;& 2x-x\ge 8+5 \Rightarrow x\ge 13.\\
\Step{3}\;& \boxed{\{x\in\R:x\ge 13\}} \ \text{(i.e., }[13,\infty)\text{)}.
\end{aligned}
\]
\end{QAPair}
